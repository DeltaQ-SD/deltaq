   51 {-----------------------------------------------------------------------------
   52     Type

   54 -- | A finite
   55 -- [signed measure](https://en.wikipedia.org/wiki/Signed_measure)
   56 -- on the number line.
   57 newtype Measure a = Measure (Piecewise (Poly a))
   58     -- INVARIANT: Adjacent pieces contain distinct objects.
   59     -- INVARIANT: The last piece is a constant polynomial,
   60     --            so that the measure is finite.
   61     deriving (Show, NFData)
   62 
   63 -- | @eval (distribution m) x@ is the measure of the interval \( (-∞, x] \).
   64 --
   65 -- This is known as the [distribution function
   66 -- ](https://en.wikipedia.org/wiki/Distribution_function_%28measure_theory%29).
   67 distribution :: (Ord a, Num a) => Measure a -> Piecewise (Poly a)
   68 distribution (Measure p) = p
   69 
   70 -- | Construct a signed measure from its
   71 -- [distribution function
   72 -- ](https://en.wikipedia.org/wiki/Distribution_function_%28measure_theory%29).
   73 --
   74 -- Return 'Nothing' if the measure is not finite,
   75 -- that is if the last piece of the piecewise function is not constant.
   76 fromDistribution
   77     :: (Ord a, Num a)
   78     => Piecewise (Poly a) -> Maybe (Measure a)
   79 fromDistribution pieces
   80     | isEventuallyConstant pieces = Just $ Measure $ trim pieces
   81     | otherwise = Nothing
   82 
   83 -- | Test whether a piecewise polynomial is consant as x -> ∞.
   84 isEventuallyConstant :: (Ord a, Num a) => Piecewise (Poly a) -> Bool
   85 isEventuallyConstant pieces
   86     | null xpolys = True
   87     | otherwise = (<= 0) . Poly.degree . snd $ last xpolys
   88   where
   89     xpolys = Piecewise.toAscPieces pieces
   90 
   91 -- | Internal.
   92 -- Join all intervals whose polynomials are equal.
   93 trim :: (Ord a, Num a) => Piecewise (Poly a) -> Piecewise (Poly a)
   94 trim = Piecewise.trim
   95 
   96 -- | Two measures are equal if they yield the same measures on every set.
   97 --
   98 -- > mx == my
   99 -- >   implies
  100 -- >   forall t. eval (distribution mx) t = eval (distribution my) t
  101 instance (Ord a, Num a) => Eq (Measure a) where
  102     Measure mx == Measure my =
  103         Piecewise.toAscPieces mx == Piecewise.toAscPieces my
  104 
  105 {-----------------------------------------------------------------------------
  106     Operations
  107 ------------------------------------------------------------------------------}
  108 -- | The measure that assigns @0@ to every set.
  109 zero :: Num a => Measure a
  110 zero = Measure Piecewise.zero
  111 
  112 -- | A
  113 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
  114 -- at the given point @x@.
  115 --
  116 -- > total (dirac x) = 1
  117 dirac :: (Ord a, Num a) => a -> Measure a
  118 dirac x = Measure $ Piecewise.fromAscPieces [(x, Poly.constant 1)]
  119 
  120 -- | The probability measure of a uniform probability distribution
  121 -- in the interval \( [x,y) \).
  122 --
  123 -- > total (uniform x y) = 1
  124 uniform :: (Ord a, Num a, Fractional a) => a -> a -> Measure a
  125 uniform x y = Measure $ case compare x y of
  126     EQ -> Piecewise.fromAscPieces [(x, 1)]
  127     _  -> Piecewise.fromAscPieces [(low, poly), (high, 1)]
  128   where
  129     low = min x y
  130     high = max x y
  131     poly = Poly.lineFromTo (low, 0) (high, 1)
  132 
  133 -- | The total of the measure applied to the set of real numbers.
  134 total :: (Ord a, Num a) => Measure a -> a
  135 total (Measure p) =
  136     case Piecewise.toAscPieces p of
  137         [] -> 0
  138         ps -> eval (snd (last ps)) 0
  139 
  140 -- | The 'support' is the smallest closed, contiguous interval \( [x,y] \)
  141 -- outside of which the measure is zero.
  142 --
  143 -- Returns 'Nothing' if the interval is empty.
  144 support :: (Ord a, Num a) => Measure a -> Maybe (a, a)
  145 support (Measure pieces) =
  146     case Piecewise.toAscPieces pieces of
  147         [] -> Nothing
  148         ps -> Just (fst $ head ps, fst $ last ps)
  149 
  150 -- | Check whether a signed measure is positive.
  151 --
  152 -- A signed measure is /positive/ if the measure of any set
  153 -- is nonnegative. In other words a positive signed measure
  154 -- is just a measure in the ordinary sense.
  155 --
  156 -- This test is nontrivial, as we have to check that the distribution
  157 -- function is monotonically increasing.
  158 isPositive :: (Ord a, Num a, Fractional a) => Measure a -> Bool
  159 isPositive (Measure m) = go 0 $ Piecewise.toAscPieces m
  160   where
  161     go _ [] =
  162         True
  163     go before ((x, o) : []) =
  164         eval before x <= eval o x
  165     go before ((x1, o) : xos@((x2, _) : _)) =
  166         (eval before x1 <= eval o x1)
  167         && Poly.isMonotonicallyIncreasingOn o (x1,x2)
  168         && go o xos
  169 
  170 {-----------------------------------------------------------------------------
  171     Operations
  172     Numerical
  173 ------------------------------------------------------------------------------}
  174 -- | Add two measures.
  175 --
  176 -- > total (add mx my) = total mx + total my
  177 add :: (Ord a, Num a) => Measure a -> Measure a -> Measure a
  178 add (Measure mx) (Measure my) =
  179     Measure $ trim $ Piecewise.zipPointwise (+) mx my
  180 
  181 -- | Scale a measure by a constant.
  182 --
  183 -- > total (scale a mx) = a * total mx
  184 scale :: (Ord a, Num a) => a -> Measure a -> Measure a
  185 scale 0 (Measure _) = zero
  186 scale x (Measure m) = Measure $ Piecewise.mapPieces (Poly.scale x) m
  187 
  188 -- | Translate a measure along the number line.
  189 --
  190 -- > eval (distribution (translate y m)) x
  191 -- >    = eval (distribution m) (x - y)
  192 translate :: (Ord a, Num a, Fractional a) => a -> Measure a -> Measure a
  193 translate y (Measure m) =
  194     Measure $ Piecewise.translateWith Poly.translate y m

  196 {-----------------------------------------------------------------------------
  197     Operations
  198     Decomposition into continuous and discrete measures,
  199     needed for convolution.
  200 ------------------------------------------------------------------------------}
  201 -- | Measure that is absolutely continuous
  202 -- with respect to the Lebesgue measure,
  203 -- Represented via its distribution function.
  204 newtype Continuous a = Continuous { unContinuous :: Piecewise (Poly a) }
  205     -- INVARIANT: The last piece is @Poly.constant p@ for some @p :: a@.
  206 
  207 -- | Density function (Radon–Nikodym derivative) of an absolutely
  208 -- continuous measure.
  209 newtype Density a = Density (Piecewise (Poly a))
  210     -- INVARIANT: The last piece is @Poly.constant 0@.
  211 
  212 -- | Density function of an absolutely continuous measure.
  213 toDensity
  214     :: (Ord a, Num a, Fractional a)
  215     => Continuous a -> Density a
  216 toDensity = Density . Piecewise.mapPieces Poly.differentiate . unContinuous
  217 
  218 -- | Decompose a mixed measure into
  219 -- a continuous measure and a discrete measure.
  220 -- See also [Lebesgue's decomposition theorem
  221 -- ](https://en.wikipedia.org/wiki/Lebesgue%27s_decomposition_theorem)
  222 decompose
  223     :: (Ord a, Num a, Fractional a)
  224     => Measure a -> (Continuous a, D.Discrete a)
  225 decompose (Measure m) =
  226     ( Continuous $ trim $ Piecewise.fromAscPieces withoutJumps
  227     , D.fromMap $ Map.fromList jumps
  228     )
  229   where
  230     pieces = Piecewise.toAscPieces m
  231 
  232     withoutJumps =
  233         zipWith (\(x,o) j -> (x, o - Poly.constant j)) pieces totalJumps
  234     totalJumps = tail $ scanl' (+) 0 $ map snd jumps
  235 
  236     jumps = go 0 pieces
  237       where
  238         go _ [] = []
  239         go prev ((x,o) : xos) =
  240             (x, Poly.eval o x - Poly.eval prev x) : go o xos
  241 
  242 {-----------------------------------------------------------------------------
  243     Observations
  244     Integration
  245 ------------------------------------------------------------------------------}
  246 -- | Integrate a polynomial @f@ with respect to the given measure @m@,
  247 -- \( \int f(x) dm(x) \).
  248 integrate :: (Ord a, Num a, Fractional a) => Poly a -> Measure a -> a
  249 integrate f m =
  250     integrateContinuous f continuous
  251     + D.integrate (eval f) discrete
  252   where
  253     (continuous, discrete) = decompose m
  254 
  255 -- | Integrate a polynomial over an absolutely continuous measure.
  256 integrateContinuous
  257     :: (Ord a, Num a, Fractional a)
  258     => Poly a -> Continuous a -> a
  259 integrateContinuous f gg
  260     | null gpieces = 0
  261     | otherwise = sum $ map integrateOverInterval $ integrands
  262   where
  263     Density g = toDensity gg
  264     gpieces = Piecewise.toAscPieces g
  265 
  266     -- Pieces on the bounded intervals
  267     boundedPieces xos =
  268         zipWith (\(x1,o) (x2,_) -> ((x1, x2), o)) xos (drop 1 xos)
  269 
  270     integrands = [ (x12, f * o) | (x12, o) <- boundedPieces gpieces ]
  271 
  272     integrateOverInterval ((x1, x2), p) =
  273         eval pp x2 - eval pp x1
  274       where
  275         pp = Poly.integrate p

  278     Operations
  279     Convolution
  280 ------------------------------------------------------------------------------}
  281 {-$ NOTE [Convolution]
  282 
  283 In order to compute a convolution,
  284 we convolve a density with the distribution function.
  285 
  286 Let $f$ denote a density, which can be continuous or a Dirac delta.
  287 Let $G$ denote a distribution function.
  288 Let $H = f * G$ be the result of the convolution.
  289 It can be shown that this is the distribution function of the
  290 convolution of the densities, $h = f * g$.
  291 
  292 The formula for convolution is
  293 
  294 $ H(y) = ∫ f(y - x) G(x) dx = ∫ f (x) G(y - x) dx$.
  295 
  296 When $f$ is a sum of delta functions, $f = Σ w_j delta_{x_j}(x)$,
  297 this integral becomes ($y - x = x_j$ => $x = y - x_j$)
  298 
  299 $ H(y) = Σ w_j G(y - x_j) $.
  300 
  301 When $f$ is a piecewise polynomial, we can convolve the pieces.
  302 
  303 When convolving with a distribution function, the final piece
  304 will be a constant $g_n$ on the interval $[x_n,∞)$.
  305 In this case, the convolution is given by
  306 
  307 \[
  308 H(y)
  309     = ∫ f (x) G(y - x) dx
  310     = ∫_{ -∞}^{y-x_n} f(x) g_n dx
  311     = g_n F(y-x_n)
  312 \]
  313 
  314 where $F$ is the distribution function of the density $f$.
  315 -}

  317 -- | Convolve a discrete measure with a mixed measure.
  318 --
  319 -- See NOTE [Convolution].
  320 convolveDiscrete
  321     :: (Ord a, Num a, Fractional a)
  322     => D.Discrete a -> Measure a -> Measure a
  323 convolveDiscrete f gg =
  324     foldr add zero
  325         [ scale w (translate x gg)
  326         | (x, w) <- Map.toAscList $ D.toMap f
  327         ]
  328 
  329 -- | Convolve an absolutely continuous measure with a mixed measure.
  330 --
  331 -- See NOTE [Convolution].
  332 convolveContinuous
  333     :: (Ord a, Num a, Fractional a)
  334     => Continuous a -> Measure a -> Measure a
  335 convolveContinuous (Continuous ff) (Measure gg)
  336     | null ffpieces = zero
  337     | null ggpieces = zero
  338     | otherwise = Measure $ trim $ boundedConvolutions + lastConvolution
  339   where
  340     ffpieces = Piecewise.toAscPieces ff
  341     ggpieces = Piecewise.toAscPieces gg
  342 
  343     Density f = toDensity (Continuous ff)
  344     fpieces = Piecewise.toAscPieces f
  345 
  346     -- Pieces on the bounded intervals
  347     boundedPieces xos =
  348         zipWith (\(x,o) (y,_) -> (x, y, o)) xos (drop 1 xos)
  349 
  350     boundedConvolutions =
  351         sum $
  352             [ Piecewise.fromAscPieces (Poly.convolve fo ggo)
  353             | fo <- boundedPieces fpieces
  354             , ggo <- boundedPieces ggpieces
  355             ]
  356 
  357     (xlast, plast) = last ggpieces
  358     glast = case Poly.toCoefficients plast of
  359         [] -> 0
  360         (a0:_) -> a0
  361     lastConvolution =
  362         Piecewise.mapPieces (Poly.scale glast)
  363         $ Piecewise.translateWith Poly.translate xlast ff
  364 
  365 -- | Additive convolution of two measures.
  366 --
  367 -- Properties:
  368 --
  369 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  370 -- >
  371 -- > convolve mx my               =  convolve my mx
  372 -- > convolve (add mx my) mz      =  add (convolve mx mz) (convolve my mz)
  373 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  374 -- > total (convolve mx my)       =  total mx * total my
  375 convolve
  376     :: (Ord a, Num a, Fractional a)
  377     => Measure a -> Measure a -> Measure a
  378 convolve mx my =
  379     add (convolveContinuous contx my) (convolveDiscrete deltax my)
  380   where
  381     (contx, deltax) = decompose mx
