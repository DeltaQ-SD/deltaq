   43 {-----------------------------------------------------------------------------
   44     Type
   45 ------------------------------------------------------------------------------}
   46 -- | A discrete, finite
   47 -- [signed measure](https://en.wikipedia.org/wiki/Signed_measure)
   48 -- on the number line.
   49 newtype Discrete a = Discrete (Map a a)
   50     -- INVARIANT: All values are non-zero.
   51     deriving (Show)
   52 
   53 -- | Internal.
   54 -- Remove all zero values.
   55 trim :: (Ord a, Num a) => Map a a -> Map a a
   56 trim m = Map.filter (/= 0) m
   57 
   58 -- | Two measures are equal if they yield the same measures on every set.
   59 --
   60 -- > mx == my
   61 -- >   implies
   62 -- >   forall t. eval (distribution mx) t = eval (distribution my) t
   63 instance (Ord a, Num a) => Eq (Discrete a) where
   64     Discrete mx == Discrete my = mx == my
   65 
   66 {-----------------------------------------------------------------------------
   67     Operations
   68 ------------------------------------------------------------------------------}
   69 -- | The measure that assigns @0@ to every set.
   70 zero :: Num a => Discrete a
   71 zero = Discrete Map.empty
   72 
   73 -- | A
   74 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
   75 -- at the given point @x@.
   76 --
   77 -- > total (dirac x) = 1
   78 dirac :: (Ord a, Num a) => a -> Discrete a
   79 dirac x = Discrete (Map.singleton x 1)
   80 
   81 -- | Construct a discrete measure
   82 -- from a collection of points and their measures.
   83 fromMap :: (Ord a, Num a) => Map a a -> Discrete a
   84 fromMap = Discrete . trim
   85 
   86 -- | Decompose the discrete measure into a collection of points
   87 -- and their measures.
   88 toMap :: Num a => Discrete a -> Map a a
   89 toMap (Discrete m) = m
   90 
   91 -- | The total of the measure applied to the set of real numbers.
   92 total :: Num a => Discrete a -> a
   93 total (Discrete m) = sum m
   94 
   95 -- | Integrate a function @f@ with respect to the given measure @m@,
   96 -- \( \int f(x) dm(x) \).
   97 integrate :: (Ord a, Num a) => (a -> a) -> Discrete a -> a
   98 integrate f (Discrete m) = sum $ Map.mapWithKey (\x w -> f x * w) m
   99 
  100 -- | @eval (distribution m) x@ is the measure of the interval \( (-âˆž, x] \).
  101 --
  102 -- This is known as the [distribution function
  103 -- ](https://en.wikipedia.org/wiki/Distribution_function_%28measure_theory%29).
  104 distribution :: (Ord a, Num a) => Discrete a -> Piecewise (Poly a)
  105 distribution (Discrete m) =
  106     Piecewise.fromAscPieces
  107     $ zipWith (\(x,_) s -> (x,Poly.constant s)) diracs steps
  108   where
  109     diracs = Map.toAscList m
  110     steps = tail $ scanl' (+) 0 $ map snd diracs
  111 
  112 -- | Add two measures.
  113 --
  114 -- > total (add mx my) = total mx + total my
  115 add :: (Ord a, Num a) => Discrete a -> Discrete a -> Discrete a
  116 add (Discrete mx) (Discrete my) =
  117     Discrete $ trim $ Map.unionWith (+) mx my
  118 
  119 -- | Scale a measure by a constant.
  120 --
  121 -- > total (scale a mx) = a * total mx
  122 scale :: (Ord a, Num a) => a -> Discrete a -> Discrete a
  123 scale 0 (Discrete _) = Discrete Map.empty
  124 scale s (Discrete m) = Discrete $ Map.map (s *) m
  125 
  126 -- | Translate a measure along the number line.
  127 --
  128 -- > eval (distribution (translate y m)) x
  129 -- >    = eval (distribution m) (x - y)
  130 translate :: (Ord a, Num a) => a -> Discrete a -> Discrete a
  131 translate y (Discrete m) = Discrete $ Map.mapKeys (y +) m
  132 
  133 -- | Additive convolution of two measures.
  134 --
  135 -- Properties:
  136 --
  137 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  138 convolve :: (Ord a, Num a) => Discrete a -> Discrete a -> Discrete a
  139 -- >
  140 -- > convolve mx my               =  convolve my mx
  141 -- > convolve (add mx my) mz      =  add (convolve mx mz) (convolve my mz)
  142 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  143 -- > total (convolve mx my)       =  total mx * total myconvolve :: (Ord a, Num a) => Discrete a -> Discrete a -> Discrete a
  144 convolve (Discrete mx) (Discrete my) =
  145     Discrete $ trim $ Map.fromListWith (+)
  146         [ (x + y, wx * wy)
  147         | (x,wx) <- Map.toList mx
  148         , (y,wy) <- Map.toList my
  149         ]
