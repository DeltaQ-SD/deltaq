   50 {-----------------------------------------------------------------------------
   51     Type
   52 ------------------------------------------------------------------------------}
   53 -- | A
   54 -- [probability measure](https://en.wikipedia.org/wiki/Probability_measure)
   55 -- on the number line.
   56 --
   57 -- A probability measure is a 'M.Measure' whose 'M.total' is @1@.
   58 newtype Prob a = Prob (Measure a)
   59     -- INVARIANT: 'M.isPositive' equals 'True'.
   60     -- INVARIANT: 'M.total' equals 1
   61     deriving (Show, NFData)
   62 
   63 -- | View the probability measure as a 'M.Measure'.
   64 measure :: (Ord a, Num a) => Prob a -> Measure a
   65 measure (Prob m) = m
   66 
   67 -- | View a 'M.Measure' as a probability distribution.
   68 --
   69 -- The measure @m@ must be positive, with total weight @1@, that is
   70 --
   71 -- > isPositive m == True
   72 -- > total m == 1
   73 --
   74 -- These preconditions are checked and the function returns 'Nothing'
   75 -- if they fail. 
   76 fromMeasure :: (Ord a, Num a, Fractional a) => Measure a -> Maybe (Prob a)
   77 fromMeasure m
   78     | M.isPositive m && M.total m == 1 = Just $ Prob m
   79     | otherwise = Nothing
   80 
   81 -- | View a 'M.Measure' as a probability distribution.
   82 --
   83 -- Variant of 'fromMeasure' where /the precondition are not checked!/
   84 unsafeFromMeasure :: Measure a -> Prob a
   85 unsafeFromMeasure = Prob
   86 
   87 -- | @eval (distribution m) x@ is the probability of picking a number @<= x@.
   88 --
   89 -- This is known as the
   90 -- [cumulative distribution function
   91 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   92 distribution :: (Ord a, Num a) => Prob a -> Piecewise (Poly a)
   93 distribution (Prob m) = M.distribution m
   94 
   95 -- | Construct a probability distribution from its
   96 -- [cumulative distribution function
   97 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   98 --
   99 -- Return 'Nothing' if
  100 -- * the cumulative distribution function is not monotonicall increasing
  101 -- * the last piece of the piecewise function is not a constant
  102 --   equal to @1@.
  103 fromDistribution
  104     :: (Ord a, Num a, Fractional a)
  105     => Piecewise (Poly a) -> Maybe (Prob a)
  106 fromDistribution pieces
  107     | Just m <- M.fromDistribution pieces = fromMeasure m
  108     | otherwise = Nothing
  109 
  110 -- | Two probability measures are equal if they have the same cumulative
  111 -- distribution functions.
  112 --
  113 -- > px == py
  114 -- >   implies
  115 -- >   forall t. eval (distribution px) t = eval (distribution py) t
  116 instance (Ord a, Num a) => Eq (Prob a) where
  117     Prob mx == Prob my = mx == my
  118 
  119 {-----------------------------------------------------------------------------
  120     Construction
  121 ------------------------------------------------------------------------------}
  122 -- | A
  123 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
  124 -- at the given point @x@.
  125 --
  126 -- @dirac x@ is the probability distribution where @x@ occurs with certainty.
  127 dirac :: (Ord a, Num a) => a -> Prob a
  128 dirac = Prob . M.dirac
  129 
  130 -- | The uniform probability distribution on the interval \( [x,y) \).
  131 uniform :: (Ord a, Num a, Fractional a) => a -> a -> Prob a
  132 uniform x y = Prob $ M.uniform x y
  133 
  134 {-----------------------------------------------------------------------------
  135     Construction
  136 ------------------------------------------------------------------------------}
  137 -- | The 'support' is the smallest closed, contiguous interval \( [x,y] \)
  138 -- outside of which the probability is zero.
  139 --
  140 -- Returns 'Nothing' if the interval is empty.
  141 support :: (Ord a, Num a) => Prob a -> Maybe (a, a)
  142 support (Prob m) = M.support m
  143 
  144 -- | Compute the
  145 -- [expected value](https://en.wikipedia.org/wiki/Expected_value)
  146 -- of a polynomial @f@ with respect to the given probability distribution,
  147 -- \( E[f(X)] \).
  148 expectation :: (Ord a, Num a, Fractional a) => Poly a -> Prob a -> a
  149 expectation f (Prob m) = M.integrate f m
  150 
  151 -- | Compute the first four
  152 -- commonly used moments of a probability distribution.
  153 moments :: (Ord a, Num a, Fractional a) => Prob a -> Moments a
  154 moments m =
  155     fromExpectedPowers (ex 1, ex 2, ex 3, ex 4)
  156   where
  157     ex n = expectation (Poly.monomial n 1) m
  158 
  159 {-----------------------------------------------------------------------------
  160     Operations
  161 ------------------------------------------------------------------------------}
  162 -- | Left-biased random choice.
  163 --
  164 -- @choice p@ is a probability distribution where
  165 -- events from the left argument are chosen with probablity @p@
  166 -- and events from the right argument are chosen with probability @(1-p)@.
  167 --
  168 -- > eval (distribution (choice p mx my)) z
  169 -- >    = p * eval (distribution mx) z + (1-p) * eval (distribution my) z
  170 choice :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a -> Prob a
  171 choice p (Prob mx) (Prob my) = Prob $
  172     M.add (M.scale p mx) (M.scale (1 - p) my)
  173 
  174 -- | Translate a probability distribution along the number line.
  175 --
  176 -- > eval (distribution (translate y m)) x
  177 -- >    = eval (distribution m) (x - y)
  178 translate :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a
  179 translate y (Prob m) = Prob $ M.translate y m
  180 
  181 -- | Additive convolution of two probability measures.
  182 --
  183 -- Properties:
  184 --
  185 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  186 -- >
  187 -- > convolve mx my               =  convolve my mx
  188 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  189 convolve
  190     :: (Ord a, Num a, Fractional a)
  191     => Prob a -> Prob a -> Prob a
  192 convolve (Prob mx) (Prob my) = Prob $ M.convolve mx my
