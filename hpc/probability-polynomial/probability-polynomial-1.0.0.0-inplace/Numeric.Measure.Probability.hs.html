   42 {-----------------------------------------------------------------------------
   43     Type
   44 ------------------------------------------------------------------------------}
   45 -- | A
   46 -- [probability measure](https://en.wikipedia.org/wiki/Probability_measure)
   47 -- on the number line.
   48 --
   49 -- A probability measure is a 'M.Measure' whose 'M.total' is @1@.
   50 newtype Prob a = Prob (Measure a)
   51     -- INVARIANT: 'M.isPositive' equals 'True'.
   52     -- INVARIANT: 'M.total' equals 1
   53     deriving (Show, NFData)
   54 
   55 -- | View the probability measure as a 'M.Measure'.
   56 measure :: (Ord a, Num a) => Prob a -> Measure a
   57 measure (Prob m) = m
   58 
   59 -- | View a 'M.Measure' as a probability distribution.
   60 --
   61 -- The measure @m@ must be positive, with total weight @1@, that is
   62 --
   63 -- > isPositive m == True
   64 -- > total m == 1
   65 --
   66 -- These preconditions are checked and the function returns 'Nothing'
   67 -- if they fail. 
   68 fromMeasure :: (Ord a, Num a, Fractional a) => Measure a -> Maybe (Prob a)
   69 fromMeasure m
   70     | M.isPositive m && M.total m == 1 = Just $ Prob m
   71     | otherwise = Nothing
   72 
   73 -- | View a 'M.Measure' as a probability distribution.
   74 --
   75 -- Variant of 'fromMeasure' where /the precondition are not checked!/
   76 unsafeFromMeasure :: Measure a -> Prob a
   77 unsafeFromMeasure = Prob
   78 
   79 -- | @eval (distribution m) x@ is the probability of picking a number @<= x@.
   80 --
   81 -- This is known as the
   82 -- [cumulative distribution function
   83 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   84 distribution :: (Ord a, Num a) => Prob a -> Piecewise (Poly a)
   85 distribution (Prob m) = M.distribution m
   86 
   87 -- | Construct a probability distribution from its
   88 -- [cumulative distribution function
   89 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   90 --
   91 -- Return 'Nothing' if
   92 -- * the cumulative distribution function is not monotonicall increasing
   93 -- * the last piece of the piecewise function is not a constant
   94 --   equal to @1@.
   95 fromDistribution
   96     :: (Ord a, Num a, Fractional a)
   97     => Piecewise (Poly a) -> Maybe (Prob a)
   98 fromDistribution pieces
   99     | Just m <- M.fromDistribution pieces = fromMeasure m
  100     | otherwise = Nothing
  101 
  102 -- | Two probability measures are equal if they have the same cumulative
  103 -- distribution functions.
  104 --
  105 -- > px == py
  106 -- >   implies
  107 -- >   forall t. eval (distribution px) t = eval (distribution py) t
  108 instance (Ord a, Num a) => Eq (Prob a) where
  109     Prob mx == Prob my = mx == my
  110 
  111 {-----------------------------------------------------------------------------
  112     Operations
  113 ------------------------------------------------------------------------------}
  114 -- | A
  115 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
  116 -- at the given point @x@.
  117 --
  118 -- @dirac x@ is the probability distribution where @x@ occurs with certainty.
  119 dirac :: (Ord a, Num a) => a -> Prob a
  120 dirac = Prob . M.dirac
  121 
  122 -- | The uniform probability distribution on the interval \( [x,y) \).
  123 uniform :: (Ord a, Num a, Fractional a) => a -> a -> Prob a
  124 uniform x y = Prob $ M.uniform x y
  125 
  126 -- | The 'support' is the smallest closed, contiguous interval \( [x,y] \)
  127 -- outside of which the probability is zero.
  128 --
  129 -- Returns 'Nothing' if the interval is empty.
  130 support :: (Ord a, Num a) => Prob a -> Maybe (a, a)
  131 support (Prob m) = M.support m
  132

  136 -- | Left-biased random choice.
  137 --
  138 -- @choice p@ is a probability distribution where
  139 -- events from the left argument are chosen with probablity @p@
  140 -- and events from the right argument are chosen with probability @(1-p)@.
  141 --
  142 -- > eval (distribution (choice p mx my)) z
  143 -- >    = p * eval (distribution mx) z + (1-p) * eval (distribution my) z
  144 choice :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a -> Prob a
  145 choice p (Prob mx) (Prob my) = Prob $
  146     M.add (M.scale p mx) (M.scale (1 - p) my)
  147 
  148 -- | Translate a probability distribution along the number line.
  149 --
  150 -- > eval (distribution (translate y m)) x
  151 -- >    = eval (distribution m) (x - y)
  152 translate :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a
  153 translate y (Prob m) = Prob $ M.translate y m
  154 
  155 -- | Additive convolution of two probability measures.
  156 --
  157 -- Properties:
  158 --
  159 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  160 -- >
  161 -- > convolve mx my               =  convolve my mx
  162 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  163 convolve
  164     :: (Ord a, Num a, Fractional a)
  165     => Prob a -> Prob a -> Prob a
  166 convolve (Prob mx) (Prob my) = Prob $ M.convolve mx my
