   51 {-----------------------------------------------------------------------------
   52     Type
   53 ------------------------------------------------------------------------------}
   54 -- | A
   55 -- [probability measure](https://en.wikipedia.org/wiki/Probability_measure)
   56 -- on the number line.
   57 --
   58 -- A probability measure is a 'M.Measure' whose 'M.total' is @1@.
   59 newtype Prob a = Prob (Measure a)
   60     -- INVARIANT: 'M.isPositive' equals 'True'.
   61     -- INVARIANT: 'M.total' equals 1
   62     deriving (Show, NFData)
   63 
   64 -- | View the probability measure as a 'M.Measure'.
   65 measure :: (Ord a, Num a) => Prob a -> Measure a
   66 measure (Prob m) = m
   67 
   68 -- | View a 'M.Measure' as a probability distribution.
   69 --
   70 -- The measure @m@ must be positive, with total weight @1@, that is
   71 --
   72 -- > isPositive m == True
   73 -- > total m == 1
   74 --
   75 -- These preconditions are checked and the function returns 'Nothing'
   76 -- if they fail. 
   77 fromMeasure :: (Ord a, Num a, Fractional a) => Measure a -> Maybe (Prob a)
   78 fromMeasure m
   79     | M.isPositive m && M.total m == 1 = Just $ Prob m
   80     | otherwise = Nothing
   81 
   82 -- | View a 'M.Measure' as a probability distribution.
   83 --
   84 -- Variant of 'fromMeasure' where /the precondition are not checked!/
   85 unsafeFromMeasure :: Measure a -> Prob a
   86 unsafeFromMeasure = Prob
   87 
   88 -- | @eval (distribution m) x@ is the probability of picking a number @<= x@.
   89 --
   90 -- This is known as the
   91 -- [cumulative distribution function
   92 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   93 distribution :: (Ord a, Num a) => Prob a -> Piecewise (Poly a)
   94 distribution (Prob m) = M.distribution m
   95 
   96 -- | Construct a probability distribution from its
   97 -- [cumulative distribution function
   98 -- ](https://en.wikipedia.org/wiki/Cumulative_distribution_function).
   99 --
  100 -- Return 'Nothing' if
  101 -- * the cumulative distribution function is not monotonicall increasing
  102 -- * the last piece of the piecewise function is not a constant
  103 --   equal to @1@.
  104 fromDistribution
  105     :: (Ord a, Num a, Fractional a)
  106     => Piecewise (Poly a) -> Maybe (Prob a)
  107 fromDistribution pieces
  108     | Just m <- M.fromDistribution pieces = fromMeasure m
  109     | otherwise = Nothing
  110 
  111 -- | Two probability measures are equal if they have the same cumulative
  112 -- distribution functions.
  113 --
  114 -- > px == py
  115 -- >   implies
  116 -- >   forall t. eval (distribution px) t = eval (distribution py) t
  117 instance (Ord a, Num a) => Eq (Prob a) where
  118     Prob mx == Prob my = mx == my
  119 
  120 {-----------------------------------------------------------------------------
  121     Construction
  122 ------------------------------------------------------------------------------}
  123 -- | A
  124 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
  125 -- at the given point @x@.
  126 --
  127 -- @dirac x@ is the probability distribution where @x@ occurs with certainty.
  128 dirac :: (Ord a, Num a) => a -> Prob a
  129 dirac = Prob . M.dirac
  130 
  131 -- | The uniform probability distribution on the interval \( [x,y) \).
  132 uniform :: (Ord a, Num a, Fractional a) => a -> a -> Prob a
  133 uniform x y = Prob $ M.uniform x y
  134 
  135 {-----------------------------------------------------------------------------
  136     Construction
  137 ------------------------------------------------------------------------------}
  138 -- | The 'support' is the smallest closed, contiguous interval \( [x,y] \)
  139 -- outside of which the probability is zero.
  140 --
  141 -- Returns 'Nothing' if the interval is empty.
  142 support :: (Ord a, Num a) => Prob a -> Maybe (a, a)
  143 support (Prob m) = M.support m
  144 
  145 -- | Compute the
  146 -- [expected value](https://en.wikipedia.org/wiki/Expected_value)
  147 -- of a polynomial @f@ with respect to the given probability distribution,
  148 -- \( E[f(X)] \).
  149 expectation :: (Ord a, Num a, Fractional a) => Poly a -> Prob a -> a
  150 expectation f (Prob m) = M.integrate f m
  151 
  152 -- | Compute the first four
  153 -- commonly used moments of a probability distribution.
  154 moments :: (Ord a, Num a, Fractional a) => Prob a -> Moments a
  155 moments m =
  156     fromExpectedPowers (ex 1, ex 2, ex 3, ex 4)
  157   where
  158     ex n = expectation (Poly.monomial n 1) m
  159

  162 ------------------------------------------------------------------------------}
  163 -- | Left-biased random choice.
  164 --
  165 -- @choice p@ is a probability distribution where
  166 -- events from the left argument are chosen with probablity @p@
  167 -- and events from the right argument are chosen with probability @(1-p)@.
  168 --
  169 -- > eval (distribution (choice p mx my)) z
  170 -- >    = p * eval (distribution mx) z + (1-p) * eval (distribution my) z
  171 choice :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a -> Prob a
  172 choice p (Prob mx) (Prob my) = Prob $
  173     M.add (M.scale p mx) (M.scale (1 - p) my)
  174 
  175 -- | Translate a probability distribution along the number line.
  176 --
  177 -- > eval (distribution (translate y m)) x
  178 -- >    = eval (distribution m) (x - y)
  179 translate :: (Ord a, Num a, Fractional a) => a -> Prob a -> Prob a
  180 translate y (Prob m) = Prob $ M.translate y m
  181 
  182 -- | Additive convolution of two probability measures.
  183 --
  184 -- Properties:
  185 --
  186 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  187 -- >
  188 -- > convolve mx my               =  convolve my mx
  189 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  190 convolve
  191     :: (Ord a, Num a, Fractional a)
  192     => Prob a -> Prob a -> Prob a
  193 convolve (Prob mx) (Prob my) = Prob $ M.convolve mx my
