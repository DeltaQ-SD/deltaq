   17 -- | The first four commonly used moments of a probability distribution.
   18 data Moments a = Moments
   19     { mean :: a
   20     -- ^ [Mean or Expected Value](https://en.wikipedia.org/wiki/Expected_value)
   21     -- \( \mu \).
   22     -- Defined as \( \mu = E[X] \).
   23     , variance :: a
   24     -- ^ [Variance](https://en.wikipedia.org/wiki/Variance) \( \sigma^2 \).
   25     -- Defined as \( \sigma^2 = E[(X - \mu)^2] \).
   26     -- Equal to \( \sigma^2 = E[X^2] - \mu^2 \).
   27     , skewness :: a
   28     -- ^ [Skewness](https://en.wikipedia.org/wiki/Skewness) \( \gamma_1 \).
   29     -- Defined as
   30     -- \( \gamma_1 = E\left[\left(\frac{(X - \mu)}{\sigma}\right)^3 \right] \).
   31     , kurtosis :: a
   32     -- ^ [Kurtosis](https://en.wikipedia.org/wiki/Kurtosis) \( \kappa \).
   33     -- Defined as
   34     --  \( \kappa = E\left[\left(\frac{(X - \mu)}{\sigma}\right)^4 \right] \).
   35     --
   36     -- The kurtosis is bounded below: \( \kappa \geq \gamma_1^2 + 1 \).
   37     }
   38     deriving (Eq, Show)
   39 
   40 -- | Compute the 'Moments' of a probability distribution given
   41 -- the expectation values of the first four powers \( m_k = E[X^k] \).
   42 --
   43 -- > fromExpectedPowers (m1,m2,m3,m4)
   44 fromExpectedPowers
   45     :: (Ord a, Num a, Fractional a)
   46     => (a, a, a, a) -> Moments a
   47 fromExpectedPowers (mean, m2, m3, m4)
   48     | variance == 0 =
   49         Moments{mean, variance, skewness = 0, kurtosis = 1}
   50     | otherwise =
   51         Moments{mean, variance, skewness, kurtosis}
   52   where
   53     meanSq = mean * mean
   54 
   55     variance = m2 - meanSq
   56     sigma = squareRoot variance
   57 
   58     skewness =
   59         (m3 - 3 * mean * variance - mean * meanSq
   60         ) / (sigma * variance)
   61 
   62     kurtosis =
   63         (m4
   64             - 4 * mean * skewness * sigma * variance
   65             - 6 * meanSq * variance
   66             - meanSq * meanSq
   67         ) / (variance * variance)
   68 
   69 -- | Helper function to approximate the square root.
   70 -- Precision: 1e-4 of the given value.
   71 --
   72 -- Uses Heron's iterative method.
   73 squareRoot :: (Ord a, Num a, Fractional a) => a -> a
   74 squareRoot x
   75     | x < 0 = error "Negative square root input"
   76     | x == 0 = 0
   77     | otherwise = goRoot x0
   78   where
   79     precision = x / 10000
   80     x0 = x/2 -- initial guess
   81     goRoot xi
   82         | abs (x - xi * xi) <= precision = xi
   83         | otherwise = goRoot ((xi + x / xi)/2)
