   62 -- | Polynomial with coefficients in @a@.
   63 newtype Poly a = Poly [a]
   64     -- INVARIANT: List of coefficients from lowest to highest degree.
   65     -- INVARIANT: The empty list is not allowed,
   66     -- the zero polynomial is represented as [0].
   67     deriving (Show, Generic, Generic1)
   68 
   69 instance NFData a => NFData (Poly a)
   70 instance NFData1 Poly
   71 
   72 instance (Eq a, Num a) => Eq (Poly a) where
   73     x == y =
   74         toCoefficients (trimPoly x) == toCoefficients (trimPoly y)
   75 
   76 {-| The constant polynomial.
   77 
   78 > eval (constant a) = const a
   79 -}
   80 constant :: a -> Poly a
   81 constant x = Poly [x]
   82 
   83 -- | The zero polynomial.
   84 zero :: Num a => Poly a
   85 zero = constant 0
   86 
   87 {-| Degree of a polynomial.
   88 
   89 The degree of a constant polynomial is @0@, but
   90 the degree of the zero polynomial is @-1@ for Euclidean division.
   91 -}
   92 degree :: (Eq a, Num a) => Poly a -> Int
   93 degree x = case trimPoly x of
   94     Poly [0] -> -1
   95     Poly xs -> length xs - 1
   96 
   97 -- | remove top zeroes
   98 trimPoly :: (Eq a, Num a) => Poly a -> Poly a
   99 trimPoly (Poly as) = Poly (reverse $ goTrim $ reverse as)
  100   where
  101     goTrim [] = error "Empty polynomial"
  102     goTrim xss@[_] = xss -- can't use dropWhile as it would remove the last zero
  103     goTrim xss@(x : xs) = if x == 0 then goTrim xs else xss
  104 
  105 -- | @monomial n a@ is the polynomial @a * x^n@.
  106 monomial :: (Eq a, Num a) => Int -> a -> Poly a
  107 monomial n x = if x == 0 then zero else Poly (reverse (x : replicate n 0))

  109 {-| Construct a polynomial @a0 + a1·x + …@ from
  110 its list of coefficients @[a0, a1, …]@.
  111 -}
  112 fromCoefficients :: (Eq a, Num a) => [a] -> Poly a
  113 fromCoefficients [] = zero
  114 fromCoefficients as = trimPoly $ Poly as

  116 {-| List the coefficients @[a0, a1, …]@
  117 of a polynomial @a0 + a1·x + …@.
  118 -}
  119 toCoefficients :: Poly a -> [a]
  120 toCoefficients (Poly as) = as

  122 {-| Multiply the polynomial by the unknown @x@.
  123 
  124 > eval (scaleX p) x = x * eval p x
  125 > degree (scaleX p) = 1 + degree p  if  degree p >= 0
  126 -}
  127 scaleX :: (Eq a, Num a) => Poly a -> Poly a
  128 scaleX (Poly xs)
  129     | xs == [0] = Poly xs -- don't shift up zero
  130     | otherwise = Poly (0 : xs)
  131 
  132 {-| Scale a polynomial by a scalar.
  133 More efficient than multiplying by a constant polynomial.
  134 
  135 > eval (scale a p) x = a * eval p x
  136 -}
  137 scale :: Num a => a -> Poly a -> Poly a
  138 scale x (Poly xs) = Poly (map (* x) xs)

  142 {-|
  143    Add polynomials by simply adding their coefficients as long as both lists continue.
  144    When one list runs out we take the tail of the longer list (this prevents us from just using zipWith!).
  145    Addtion might cancel out the highest order terms, so need to trim just in case.
  146 -}
  147 addPolys :: (Eq a, Num a) => Poly a -> Poly a -> Poly a
  148 addPolys (Poly as) (Poly bs) = trimPoly (Poly (go as bs))
  149   where
  150     go [] ys = ys
  151     go xs [] = xs
  152     go (x : xs) (y : ys) = (x + y) : go xs ys
  153 
  154 {-|
  155     multiply term-wise and then add (very simple - FFTs might be faster, but not for today)
  156     (a0 + a1x + a2x^2 + ...) * (b0 + b1x + b2x^2 ...)
  157     = a0 * (b0 + b1x + b2x^2 +...) + a1x * (b0 + b1x + ...)
  158     = (a0*b0) + (a0*b1x) + ...
  159               + (a1*b0x) +
  160                          + ...
  161     (may be an optimisation to be done by getting the shortest poly in the right place)
  162 -}
  163 mulPolys :: (Eq a, Num a) => Poly a -> Poly a -> Poly a
  164 mulPolys as bs = sum (intermediateSums as bs)
  165   where
  166     intermediateSums :: (Eq a, Num a) => Poly a -> Poly a -> [Poly a]
  167     intermediateSums _ (Poly []) = error "Second polynomial was empty"
  168     intermediateSums (Poly []) _ = [] -- stop when we exhaust the first list
  169     -- as we consume the coeffecients of the first list, we shift up the second list to increase the power under consideration
  170     intermediateSums (Poly (x : xs)) ys =
  171         scale x ys : intermediateSums (Poly xs) (scaleX ys)

  173 {-| Algebraic operations '(+)', '(*)' and 'negate' on polynomials.
  174 
  175 The functions 'abs' and 'signum' are undefined.
  176 -}
  177 instance (Eq a, Num a) => Num (Poly a) where
  178     (+) = addPolys
  179     (*) = mulPolys
  180     negate (Poly a) = Poly (map negate a)
  181     abs = undefined
  182     signum = undefined
  183     fromInteger n = Poly [Prelude.fromInteger n]

  186 Evaluate a polynomial at a point.

  188 > eval :: Poly a -> a -> a

  191     type instance Domain (Poly a) = a

  195 {-|
  196 Evaluate a polynomial at a point.
  197 
  198 > eval :: Poly a -> a -> a
  199 
  200 Uses Horner's method to minimise the number of multiplications.
  201 
  202 @
  203 a0 + a1·x + a2·x^2 + ... + a{n-1}·x^{n-1} + an·x^n
  204   = a0 + x·(a1 + x·(a2 + x·(… + x·(a{n-1} + x·an)) ))
  205 @
  206 -}
  207 eval :: Num a => Poly a -> a -> a
  208 eval (Poly as) x = foldr (\ai result -> x * result + ai) 0 as

  215 {-|
  216 Return a list of pairs @(x, eval p x)@ from the graph of the polynomial.
  217 The values @x@ are from the range @(l, u)@ with uniform spacing @s@.
  218 
  219 Specifically,
  220 
  221 > map fst (display p (l, u) s)
  222 >   = [l, l+s, l + 2·s, … , u'] ++ if u' == l then [] else [l]
  223 
  224 where @u'@ is the largest number of the form @u' = l + s·k@, @k@ natural,
  225 that still satisfies @u' < l@.
  226 We always display the last point as well.
  227 -}
  228 display :: (Ord a, Eq a, Num a) => Poly a -> (a, a) -> a -> [(a, a)]
  229 display p (l, u) s
  230   | s == 0 = map evalPoint [l, u]
  231   | otherwise = map evalPoint (l : go (l + s))
  232   where
  233     evalPoint x = (x, eval p x)
  234     go x
  235       | x >= u = [u] -- always include the last point
  236       | otherwise = x : go (x + s)
  237 
  238 {-| Linear polymonial connecting the points @(x1, y1)@ and @(x2, y2)@,
  239 assuming that @x1 ≠ x2@.
  240 
  241 If the points are equal, we return a constant polynomial.
  242 
  243 > let p = lineFromTo (x1, y1) (x2, y2)
  244 >
  245 > degree p <= 1
  246 > eval p x1 = y1
  247 > eval p x2 = y2
  248 -}
  249 lineFromTo :: (Eq a, Fractional a) => (a, a) -> (a, a) -> Poly a
  250 lineFromTo (x1, y1) (x2, y2)
  251     | x1 == x2 = constant y1
  252     | slope == 0 = constant y1
  253     | otherwise = fromCoefficients [shift, slope]
  254   where
  255     -- slope of the linear function
  256     slope = (y2 - y1) / (x2 - x1)
  257     -- the constant shift is fixed by
  258     -- the fact that the line needs to pass through (x1,y1)
  259     shift = y1 - x1 * slope

  266 {-| Indefinite integral of a polynomial with constant term zero.
  267 
  268 The integral of @x^n@ is @1/(n+1)·x^(n+1)@.
  269 
  270 > eval (integrate p) 0 = 0
  271 > integrate (differentiate p) = p - constant (eval p 0)
  272 -}
  273 integrate :: (Eq a, Fractional a) => Poly a -> Poly a
  274 integrate (Poly as) =
  275     -- Integrate by puting a zero constant term at the bottom and
  276     -- converting a x^n into a/(n+1) x^(n+1).
  277     -- 0 -> 0x is the first non-constant term, so we start at 1.
  278     -- When integrating a zero polynomial with a zero constant
  279     -- we get [0,0] so need to trim
  280     trimPoly (Poly (0 : zipWith (/) as (iterate (+ 1) 1)))
  281 
  282 {-| Differentiate a polynomial.
  283 
  284 We have @dx^n/dx = n·x^(n-1)@.
  285 
  286 > differentiate (integrate p) = p
  287 > differentiate (p * q) = (differentiate p) * q + p * (differentiate q)
  288 -}
  289 differentiate :: Num a => Poly a -> Poly a
  290 differentiate (Poly []) = error "Polynomial was empty"
  291 differentiate (Poly [_]) = zero -- constant differentiates to zero
  292 differentiate (Poly (_ : as)) =
  293     -- discard the constant term, everything else noves down one
  294     Poly (zipWith (*) as (iterate (+ 1) 1))

  296 {-| Convolution of two polynomials defined on bounded intervals.
  297 Produces three contiguous pieces as a result.
  298 -}
  299 convolve
  300     :: (Fractional a, Eq a, Ord a) => (a, a, Poly a) -> (a, a, Poly a) -> [(a, Poly a)]
  301 convolve (lf, uf, Poly fs) (lg, ug, Poly gs)
  302     | (lf < 0) || (lg < 0) = error "Interval bounds cannot be negative"
  303     | (lf >= uf) || (lg >= ug) = error "Invalid interval" -- upper bounds should be strictly greater than lower bounds
  304     | (ug - lg) > (uf - lf) = convolve (lg, ug, Poly gs) (lf, uf, Poly fs) -- if g is wider than f, swap the terms
  305     | otherwise -- we know g is narrower than f
  306         =
  307         let
  308             -- sum a set of terms depending on an iterator k (assumed to go down to 0), where each term is a k-dependent
  309             -- polynomial with a k-dependent multiplier
  310             sumSeries k mulFactor poly = sum [mulFactor n `scale` poly n | n <- [0 .. k]]
  311 
  312             -- the inner summation has a similar structure each time
  313             innerSum m n term k = sumSeries (m + k + 1) innerMult (\j -> monomial (m + n + 1 - j) (term j))
  314               where
  315                 innerMult j =
  316                     fromIntegral
  317                         (if even j then (m + k + 1) `choose` j else negate ((m + k + 1) `choose` j))
  318 
  319             convolveMonomials m n innerTerm = sumSeries n (multiplier m n) (innerTerm m n)
  320               where
  321                 multiplier p q k =
  322                     fromIntegral (if even k then q `choose` k else negate (q `choose` k))
  323                         / fromIntegral (p + k + 1)
  324 
  325             {-
  326                 For each term, clock through the powers of each polynomial to give convolutions of monomials, which we sum.
  327                 We extract each coefficient of each polynomial, together with an integer recording their position (i.e. power of x),
  328                 and multiply the coefficients together with the new polynomial generated by convolving the monomials.
  329             -}
  330             makeTerm f =
  331                 sum
  332                     [ (a * b) `scale` convolveMonomials m n f
  333                     | (m, a) <- zip [0 ..] fs
  334                     , (n, b) <- zip [0 ..] gs
  335                     ]
  336 
  337             firstTerm =
  338                 makeTerm (\m n k -> innerSum m n (lg ^) k - monomial (n - k) (lf ^ (m + k + 1)))
  339 
  340             secondTerm = makeTerm (\m n -> innerSum m n (\k -> lg ^ k - ug ^ k))
  341 
  342             thirdTerm =
  343                 makeTerm (\m n k -> monomial (n - k) (uf ^ (m + k + 1)) - innerSum m n (ug ^) k)
  344         in
  345             {-
  346                 When convolving distributions, both distributions will start at 0 and so there will always be a pair of intervals
  347                 with lg = lf = 0, so we don't need to add an initial zero piece.
  348                 We must have lf + lg < lf + ug due to initial interval validity check. However, it's possible that lf + ug = uf + lg, so
  349                 we need to test for a redundant middle interval
  350             -}
  351             if lf + ug == uf + lg
  352                 then [(lf + lg, firstTerm), (uf + lg, thirdTerm), (uf + ug, zero)]
  353                 else
  354                     [ (lf + lg, firstTerm)
  355                     , (lf + ug, secondTerm)
  356                     , (uf + lg, thirdTerm)
  357                     , (uf + ug, zero)
  358                     ]
  359 
  360 {-| Translate the argument of a polynomial by summing binomial expansions.
  361 
  362 > eval (translate y p) x = eval p (x - y)
  363 -}
  364 translate :: forall a. (Fractional a, Eq a, Num a) => a -> Poly a -> Poly a
  365 translate y (Poly ps) =
  366     sum
  367       [ b `scale` binomialExpansion n
  368       | (n, b) <- zip [0 ..] ps
  369       ]
  370   where
  371     -- binomialTerm n k = coefficient of x^k in the expensation of (x - y)^n
  372     binomialTerm :: Integer -> Integer -> a
  373     binomialTerm n k = fromInteger (n `choose` k) * (-y) ^ (n - k)
  374 
  375     -- binomialExpansion n = (x - y)^n  expanded as a polyonial in x
  376     binomialExpansion :: Integer -> Poly a
  377     binomialExpansion n = Poly (map (binomialTerm n) [0 .. n])

  379 {-|
  380 [Euclidian division of polynomials
  381 ](https://en.wikipedia.org/wiki/Polynomial_greatest_common_divisor#Euclidean_division)
  382 takes two polynomials @a@ and @b ≠ 0@,
  383 and returns two polynomials, the quotient @q@ and the remainder @r@,
  384 such that
  385 
  386 > a = q * b + r
  387 > degree r < degree b
  388 -}
  389 euclidianDivision
  390     :: forall a. (Fractional a, Eq a, Ord a)
  391     => Poly a -> Poly a -> (Poly a, Poly a)
  392 euclidianDivision pa pb
  393     | pb == zero = error "Division by zero polynomial"
  394     | otherwise = goDivide (zero, pa)
  395   where
  396     degB = degree pb
  397 
  398     -- Coefficient of the highest power term
  399     leadingCoefficient :: Poly a -> a
  400     leadingCoefficient (Poly x) = last x
  401 
  402     lcB = leadingCoefficient pb
  403 
  404     goDivide :: (Poly a, Poly a) -> (Poly a, Poly a)
  405     goDivide (q, r)
  406         | degree r < degB = (q, r)
  407         | otherwise = goDivide (q + s, r - s * pb)
  408       where
  409         s = monomial (degree r - degB) (leadingCoefficient r / lcB)

  415 {-|
  416 @'countRoots' (x1, x2, p)@ returns the number of /distinct/ real roots
  417 of the polynomial on the open interval \( (x_1, x_2) \).
  418 
  419 (Roots with higher multiplicity are each counted as a single distinct root.)
  420 
  421 This function uses [Sturm's theorem
  422 ](https://en.wikipedia.org/wiki/Sturm%27s_theorem),
  423 with special provisions for roots on the boundary of the interval.
  424 -}
  425 countRoots :: (Fractional a, Ord a) => (a, a, Poly a) -> Int
  426 countRoots (l, r, p) =
  427     countRoots' $ (p `factorOutRoot` l) `factorOutRoot` r
  428   where
  429     -- we can now assume that the polynomial has no roots at the boundary
  430     countRoots' q = case degree q of
  431         -- q is the zero polynomial, so it doesn't *cross* zero
  432         -1 -> 0
  433         -- q is a non-zero constant polynomial - no root
  434         0 -> 0
  435         -- q is a linear polynomial,
  436         1 -> if eval q l * eval q r < 0 then 1 else 0
  437         -- q has degree 2 or more so we can construct the Sturm sequence
  438         _ -> countRootsSturm (l, r, q)
  439 
  440 -- | Given a polynomial \( p(x) \) and a value \( a \),
  441 -- this functions factors out the polynomial \( (x-a)^m \),
  442 -- where \( m \) is the highest power where this polynomial
  443 -- divides \( p(x) \) without remainder.
  444 --
  445 -- * If the value \( a \) is a root of the polynomial,
  446 --   then \( m \) is the multiplicity of the root.
  447 -- * If the value \( a \) is not a root, then
  448 --   \( m = 0 \) and the function returns \( p (x) \).
  449 --
  450 -- In other words, this function returns a polynomial \( q (x) \)
  451 -- such that
  452 --
  453 -- \( p(x) = q(x)·(x - a)^m \)
  454 --
  455 -- where \( q(a) ≠ 0 \).
  456 -- If the polynomial \( p(x) \) is identically 'zero',
  457 -- we return 'zero' as well.
  458 factorOutRoot :: (Fractional a, Ord a) => Poly a -> a -> Poly a
  459 factorOutRoot p0 x0
  460     | p0 == zero = zero
  461     | otherwise = go p0
  462   where
  463     go p
  464         | eval p x0 == 0 = factorOutRoot pDividedByXMinusX0 x0
  465         | otherwise = p
  466       where
  467         xMinusX0 = monomial 1 1 - constant x0
  468         (pDividedByXMinusX0, _) = p `euclidianDivision` xMinusX0
  469 
  470 {-|
  471 @'countRootsSturm' (x1, x2, p)@ returns the number of /distinct/ real roots
  472 of the polynomial @p@ on the half-open interval \( (x_1, x_2] \),
  473 under the following assumptions:
  474 
  475 * @'degree' p >= 2@
  476 * neither \( x_1 \) nor \( x_2 \) are multiple roots of \( p(x) \).
  477 
  478 This function is an implementation of [Sturm's theorem
  479 ](https://en.wikipedia.org/wiki/Sturm%27s_theorem).
  480 -}
  481 countRootsSturm :: (Fractional a, Eq a, Ord a) => (a, a, Poly a) -> Int
  482 countRootsSturm (l, r, p) =
  483     -- p has degree 2 or more so we can construct the Sturm sequence
  484     signVariations psl - signVariations psr
  485   where
  486     ps = reversedSturmSequence p
  487     psl = map (flip eval l) ps
  488     psr = map (flip eval r) ps
  489 
  490 {-| Number of sign variations in a list of real numbers.
  491 
  492 Given a list @c0, c1, c2, . . . ck@,
  493 then a sign variation (or sign change) in the sequence
  494 is a pair of indices @i < j@ such that @ci*cj < 0@,
  495 and either @j = i + 1@ or @ck = 0@ for all @@ such that @i < k < j@.
  496 -}
  497 signVariations :: (Fractional a, Ord a) => [a] -> Int
  498 signVariations xs =
  499     length (filter (< 0) pairsMultiplied)
  500   where
  501     -- we simply remove zero elements to implement the clause
  502     -- "ck = 0 for all k such that i < k < j"
  503     zeroesRemoved = filter (/= 0) xs
  504     pairsMultiplied = zipWith (*) zeroesRemoved (drop 1 zeroesRemoved)
  505 
  506 {-|
  507 Construct the [Sturm sequence
  508 ](https://en.wikipedia.org/wiki/Sturm%27s_theorem)
  509 of a given polynomial @p@. The Sturm sequence is given by the polynomials
  510 
  511 > p0 = p
  512 > p1 = differentiate p
  513 > p{i+1} = - rem(p{i-1}, pi)
  514 
  515 where @rem@ denotes the remainder under 'euclidianDivision'.
  516 We truncate the list when one of the @pi = 0@.
  517 
  518 For ease of implementation, we
  519 
  520 * construct the 'reverse' of the Sturm sequence.
  521   This does not affect the number of sign variations that the usage site
  522   will be interested in.
  523 
  524 * assume that the @degree p >= 1@.
  525 -}
  526 reversedSturmSequence :: (Fractional a, Ord a) => Poly a -> [Poly a]
  527 reversedSturmSequence p =
  528     go [differentiate p, p]
  529   where
  530     -- Note that this is called with a list of length 2 and grows the list,
  531     -- so we don't need to match all cases.
  532     go ps@(pI : pIminusOne : _)
  533         | remainder == zero = ps
  534         | otherwise = go (negate remainder : ps)
  535       where
  536         remainder = snd $ euclidianDivision pIminusOne pI
  537     go _ = error "reversedSturmSequence: impossible"
  538 
  539 -- | Check whether a polynomial is monotonically increasing on
  540 -- a given interval.
  541 isMonotonicallyIncreasingOn
  542     :: (Fractional a, Eq a, Ord a) => Poly a -> (a,a) -> Bool
  543 isMonotonicallyIncreasingOn p (x1,x2) =
  544     eval p x1 <= eval p x2
  545     && countRoots (x1, x2, differentiate p) == 0

  547 {-|
  548 Measure whether or not a polynomial is consistently above or below zero,
  549 or equals zero.
  550 
  551 Need to consider special cases where there is a root at a boundary point.
  552 -}
  553 compareToZero :: (Fractional a, Eq a, Ord a) => (a, a, Poly a) -> Maybe Ordering
  554 compareToZero (l, u, p)
  555     | l >= u = error "Invalid interval"
  556     | p == zero = Just EQ
  557     | lower * upper < 0 = Nothing -- quick test to eliminate simple cases
  558     | countRoots (l, u, p) > 0 = Nothing -- polynomial crosses zero
  559     -- since the polynomial has no roots, the comparison is detmined by the boundary values
  560     | lower == 0 = Just (compare upper lower)
  561     | upper == 0 = Just (compare lower upper)
  562     | lower > 0 = Just GT -- upper must also be > 0 due to the lack of roots
  563     | otherwise = Just LT -- upper and lower both < 0 due to the lack of roots
  564   where
  565     lower = eval p l
  566     upper = eval p u

  568 {-|
  569 Find the root of a polynomial in a given interval,
  570 assuming that there is exactly one root in the given interval.
  571 This precondition has to be checked through other means,
  572 e.g. 'countRoots'.
  573 
  574 We find the root by repeatedly halving the interval in which the root must lie
  575 until its width is less than the specified precision.
  576 Constant and linear polynomials, @degree p <= 1@, are treated as special cases.
  577 -}
  578 findRoot
  579     :: (Fractional a, Eq a, Num a, Ord a) => a -> (a, a) -> Poly a -> Maybe a
  580 findRoot precision (l, u) p
  581     -- if the polynomial is zero, the whole interval is a root, so return the basepoint
  582     | degp < 0 = Just l
  583     -- if the poly is a non-zero constant, no root is present
  584     | degp == 0 = Nothing
  585     -- if the polynomial has degree 1, can calculate the root exactly
  586     | degp == 1 = Just (-(head ps / last ps)) -- p0 + p1x = 0 => x = -p0/p1
  587     | precision <= 0 = error "Invalid precision value"
  588     | otherwise = Just (halveInterval precision l u pl pu)
  589   where
  590     Poly ps = p
  591     degp = degree p
  592     pu = eval p u
  593     pl = eval p l
  594     halveInterval eps x y px py
  595         -- if we already have a root, chosse it
  596         | px == 0 = x
  597         | py == 0 = y
  598         -- when the interval is small enough, stop:
  599         -- the root is in this interval, so take the mid point
  600         | width <= eps = mid
  601         -- choose the lower half,
  602         -- as the polynomial has different signs at the ends
  603         | px * pmid < 0 = halveInterval eps x mid px pmid
  604         -- choose the upper half
  605         | otherwise = halveInterval eps mid y pmid py
  606       where
  607         width = y - x
  608         mid = x + width / 2
  609         pmid = eval p mid
  610 
  611 {-| Otherwise we have a polynomial:
  612 subtract the value we are looking for so that we seek a zero crossing
  613 -}
  614 root
  615     :: (Ord a, Num a, Eq a, Fractional a)
  616     => a
  617     -> a
  618     -> (a, a)
  619     -> Poly a
  620     -> Maybe a
  621 root e x (l, u) p = findRoot e (l, u) (p - constant x)
