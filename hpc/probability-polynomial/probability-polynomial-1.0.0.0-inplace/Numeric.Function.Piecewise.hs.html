   48 {-----------------------------------------------------------------------------
   49     Type
   50 ------------------------------------------------------------------------------}
   51 -- | Internal representation of a single piece,
   52 -- starting at a basepoint of type @a@
   53 -- and containing an object of type @o@.
   54 data Piece a o = Piece
   55     { basepoint :: a
   56     , object :: o
   57     }
   58     deriving (Eq, Show, Generic, NFData)
   59 
   60 {- | A function defined piecewise on numerical intervals.
   61  
   62 * @o@ = type of function on every piece
   63     e.g. polynomials or other specialized representations of functions
   64 * @'Fun.Domain' o@ = numerical type for the number line, e.g. 'Rational' or 'Double'
   65 
   66 A value @f :: Piecewise o@ represents a function
   67 
   68 > eval f x = { 0           if -∞ <  x < x1
   69 >            { eval o1 x   if x1 <= x < x2
   70 >            { eval o2 x   if x2 <= x < x3
   71 >            { …
   72 >            { eval on x   if xn <= x < +∞
   73 
   74 where @x1, …, xn@ are points on the real number line
   75 (in strictly increasing order)
   76 and where @o1, …, on@ are specialized representations functions,
   77 e.g. polynomials.
   78 
   79 In other words, the value @f@ represents a function that
   80 is defined piecewise on half-open intervals.
   81 
   82 The function 'intervals' returns the half-open intervals in the middle:
   83 
   84 > intervals f = [(x1,x2), (x2,x3), …, (xn-1, xn)]
   85 
   86 No attempt is made to merge intervals if the piecewise objects are equal,
   87 e.g. the situation @o1 == o2@ may occur.
   88 
   89 -}
   90 data Piecewise o
   91     = Pieces [Piece (Fun.Domain o) o]
   92     deriving (Generic)
   93 
   94 deriving instance (Show (Fun.Domain o), Show o) => Show (Piecewise o)
   95 deriving instance (NFData (Fun.Domain o), NFData o) => NFData (Piecewise o)
   96 
   97 {-$Piecewise Invariants
   98 
   99 * The empty list represents the zero function.
  100 * The 'basepoint's are in strictly increasing order.
  101 * The internal representation of the function mentioned in the definition is
  102 
  103     > f = Pieces [Piece x1 o1, Piece x2 o2, …, Piece xn on]
  104 -}
  105 
  106 {-----------------------------------------------------------------------------
  107     Operations
  108 ------------------------------------------------------------------------------}
  109 -- | The function which is zero everywhere.
  110 zero :: Piecewise o
  111 zero = Pieces []
  112 
  113 -- | @fromInterval (x1,x2) o@ creates a 'Piecewise' function
  114 -- from a single function @o@ by restricting it to the
  115 -- to half-open interval @x1 <= x < x2@.
  116 -- The result is zero outside this interval.
  117 fromInterval
  118     :: (Ord (Fun.Domain o), Num o)
  119     => (Fun.Domain o, Fun.Domain o) -> o -> Piecewise o
  120 fromInterval (x,y) o = Pieces [Piece start o, Piece end 0]
  121   where
  122     start = min x y
  123     end = max x y
  124 
  125 -- | Build a piecewise function from an ascending list of contiguous pieces.
  126 --
  127 -- /The precondition (`map fst` of input list is ascending) is not checked./
  128 fromAscPieces :: Ord (Fun.Domain o) => [(Fun.Domain o, o)] -> Piecewise o
  129 fromAscPieces = Pieces . map (uncurry Piece)
  130 
  131 -- | Convert the piecewise function to a list of contiguous pieces
  132 -- where the starting points of the pieces are in ascending order.
  133 toAscPieces :: Ord (Fun.Domain o) => Piecewise o -> [(Fun.Domain o, o)]
  134 toAscPieces (Pieces xos) = [ (x, o) | Piece x o <- xos ]
  135 
  136 -- | Intervals on which the piecewise function is defined, in sequence.
  137 -- The last half-open interval, @xn <= x < +∞@, is omitted.
  138 intervals :: Piecewise o -> [(Fun.Domain o, Fun.Domain o)]
  139 intervals (Pieces ys) =
  140     zip (map basepoint ys) (drop 1 $ map basepoint ys)
  141 
  142 {-----------------------------------------------------------------------------
  143     Operations
  144     Structure
  145 ------------------------------------------------------------------------------}
  146 -- | Map the objects of pieces.
  147 mapPieces
  148     :: Fun.Domain o ~ Fun.Domain o'
  149     => (o -> o') -> Piecewise o -> Piecewise o'
  150 mapPieces f (Pieces ps) = Pieces [ Piece x (f o) | Piece x o <- ps ]
  151 
  152 -- | Merge all adjacent pieces whose functions are considered
  153 -- equal by the given predicate.
  154 mergeBy :: Num o => (o -> o -> Bool) -> Piecewise o -> Piecewise o
  155 mergeBy eq (Pieces pieces) = Pieces $ go 0 pieces
  156   where
  157     go _ [] = []
  158     go before (p : ps)
  159         | before `eq` object p = go before ps
  160         | otherwise = p : go (object p) ps
  161 
  162 -- | Merge all adjacent pieces whose functions are equal according to '(==)'.
  163 trim :: (Eq o, Num o) => Piecewise o -> Piecewise o
  164 trim = mergeBy (==)

  170 {-|
  171 Evaluate a piecewise function at a point.
  172 
  173 * @'Fun.Domain' ('Piecewise' o) = 'Fun.Domain' o@
  174 * @'Fun.Codomain' ('Piecewise' o) = 'Fun.Codomain' o@
  175 -}
  176 instance (Fun.Function o, Num o, Ord (Fun.Domain o), Num (Fun.Codomain o))
  177     => Fun.Function (Piecewise o)
  178   where
  179     type instance Domain (Piecewise o) = Fun.Domain o
  180     type instance Codomain (Piecewise o) = Fun.Codomain o
  181     eval = evaluate
  182 
  183 -- | Evaluate the piecewise function at a point.
  184 -- See 'Piecewise' for the semantics.
  185 evaluate
  186     :: (Fun.Function o, Num o, Ord (Fun.Domain o), Num (Fun.Codomain o))
  187     => Piecewise o -> Fun.Domain o -> Fun.Codomain o
  188 evaluate (Pieces pieces) x = go 0 pieces
  189  where
  190     go before [] = Fun.eval before x
  191     go before (p:ps)
  192         | basepoint p <= x = go (object p) ps
  193         | otherwise = Fun.eval before x
  194 
  195 -- | Translate a piecewise function,
  196 -- given a way to translate each piece.
  197 --
  198 -- >  eval (translate' y o) = eval o (x - y)
  199 -- >    implies
  200 -- >    eval (translateWith translate' y p) = eval p (x - y)
  201 translateWith
  202     :: (Ord (Fun.Domain o), Num (Fun.Domain o), Num o)
  203     => (Fun.Domain o -> o -> o)
  204     -> Fun.Domain o -> Piecewise o -> Piecewise o
  205 translateWith trans y (Pieces pieces) =
  206     Pieces [ Piece (x + y) (trans y o) | Piece x o <- pieces ]
  207 
  208 {-----------------------------------------------------------------------------
  209     Operations
  210     Zip
  211 ------------------------------------------------------------------------------}
  212 -- | Combine two piecewise functions by combining the pieces
  213 -- with a pointwise operation that preserves @0@.
  214 --
  215 -- For example, `(+)` and `(*)` are pointwise operations on functions,
  216 -- but convolution is not a pointwise operation.
  217 --
  218 -- Preconditions on the argument @f@:
  219 --
  220 -- * @f 0 0 = 0@
  221 -- * @f@ is a pointwise operations on functions,
  222 --   e.g. commutes with pointwise evaluation.
  223 --
  224 -- /The preconditions are not checked!/
  225 zipPointwise
  226     :: (Ord (Fun.Domain o), Num o)
  227     => (o -> o -> o)
  228         -- ^ @f@
  229     -> Piecewise o -> Piecewise o -> Piecewise o
  230 zipPointwise f (Pieces xs') (Pieces ys') =
  231     Pieces $ go 0 xs' 0 ys'
  232   where
  233     -- We split the intervals and combine the pieces in a single pass.
  234     --
  235     -- The algorithm is similar to mergesort:
  236     -- We walk both lists in parallel and generate a new piece by
  237     -- * taking the basepoint of the nearest piece
  238     -- * and combining it with the object that was overhanging from
  239     --   the previous piece (`xhang`, `yhang`)
  240     go _ [] _ [] = []
  241     go _ (Piece x ox : xstail) yhang [] =
  242         Piece x (f ox yhang) : go ox xstail yhang []
  243     go xhang [] _ (Piece y oy : ystail) =
  244         Piece y (f xhang oy) : go xhang [] oy ystail
  245     go xhang xs@(Piece x ox : xstail) yhang ys@(Piece y oy : ystail) =
  246         case compare x y of
  247             LT -> Piece x (f ox    yhang) : go ox xstail yhang ys
  248             EQ -> Piece x (f ox    oy   ) : go ox xstail oy ystail
  249             GT -> Piece y (f xhang oy   ) : go xhang xs  oy ystail
  250 
  251 {-----------------------------------------------------------------------------
  252     Operations
  253     Numeric
  254 ------------------------------------------------------------------------------}
  255 {-| Algebraic operations '(+)', '(*)' and 'negate' on piecewise functions.
  256 
  257 The functions 'abs' and 'signum' are defined using 'abs' and 'signum'
  258 for every piece.
  259 
  260 TODO: 'fromInteger' is __undefined__
  261 -}
  262 instance (Ord (Fun.Domain o), Num o) => Num (Piecewise o) where
  263     (+) = zipPointwise (+)
  264     (*) = zipPointwise (*)
  265     negate = mapPieces negate
  266     abs = mapPieces abs
  267     signum = mapPieces signum
  268     fromInteger 0 = zero
  269     fromInteger n = error "TODO: fromInteger not implemented"
