   49 {-----------------------------------------------------------------------------
   50     Type

   52 -- | A finite
   53 -- [signed measure](https://en.wikipedia.org/wiki/Signed_measure)
   54 -- on the number line.
   55 newtype Measure a = Measure (Piecewise (Poly a))
   56     -- INVARIANT: Adjacent pieces contain distinct objects.
   57     -- INVARIANT: The last piece is a constant polynomial,
   58     --            so that the measure is finite.
   59     deriving (Show, NFData)
   60 
   61 -- | @eval (distribution m) x@ is the measure of the interval \( (-∞, x] \).
   62 --
   63 -- This is known as the [distribution function
   64 -- ](https://en.wikipedia.org/wiki/Distribution_function_%28measure_theory%29).
   65 distribution :: (Ord a, Num a) => Measure a -> Piecewise (Poly a)
   66 distribution (Measure p) = p
   67 
   68 -- | Construct a signed measure from its
   69 -- [distribution function
   70 -- ](https://en.wikipedia.org/wiki/Distribution_function_%28measure_theory%29).
   71 --
   72 -- Return 'Nothing' if the measure is not finite,
   73 -- that is if the last piece of the piecewise function is not constant.
   74 fromDistribution
   75     :: (Ord a, Num a)
   76     => Piecewise (Poly a) -> Maybe (Measure a)
   77 fromDistribution pieces
   78     | isEventuallyConstant pieces = Just $ Measure $ trim pieces
   79     | otherwise = Nothing
   80 
   81 -- | Test whether a piecewise polynomial is consant as x -> ∞.
   82 isEventuallyConstant :: (Ord a, Num a) => Piecewise (Poly a) -> Bool
   83 isEventuallyConstant pieces
   84     | null xpolys = True
   85     | otherwise = (<= 0) . Poly.degree . snd $ last xpolys
   86   where
   87     xpolys = Piecewise.toAscPieces pieces
   88 
   89 -- | Internal.
   90 -- Join all intervals whose polynomials are equal.
   91 trim :: (Ord a, Num a) => Piecewise (Poly a) -> Piecewise (Poly a)
   92 trim = Piecewise.trim
   93 
   94 -- | Two measures are equal if they yield the same measures on every set.
   95 --
   96 -- > mx == my
   97 -- >   implies
   98 -- >   forall t. eval (distribution mx) t = eval (distribution my) t
   99 instance (Ord a, Num a) => Eq (Measure a) where
  100     Measure mx == Measure my =
  101         Piecewise.toAscPieces mx == Piecewise.toAscPieces my
  102 
  103 {-----------------------------------------------------------------------------
  104     Operations
  105 ------------------------------------------------------------------------------}
  106 -- | The measure that assigns @0@ to every set.
  107 zero :: Num a => Measure a
  108 zero = Measure Piecewise.zero
  109 
  110 -- | A
  111 -- [Dirac measure](https://en.wikipedia.org/wiki/Dirac_measure)
  112 -- at the given point @x@.
  113 --
  114 -- > total (dirac x) = 1
  115 dirac :: (Ord a, Num a) => a -> Measure a
  116 dirac x = Measure $ Piecewise.fromAscPieces [(x, Poly.constant 1)]
  117 
  118 -- | The probability measure of a uniform probability distribution
  119 -- in the interval \( [x,y) \).
  120 --
  121 -- > total (uniform x y) = 1
  122 uniform :: (Ord a, Num a, Fractional a) => a -> a -> Measure a
  123 uniform x y = Measure $ case compare x y of
  124     EQ -> Piecewise.fromAscPieces [(x, 1)]
  125     _  -> Piecewise.fromAscPieces [(low, poly), (high, 1)]
  126   where
  127     low = min x y
  128     high = max x y
  129     poly = Poly.lineFromTo (low, 0) (high, 1)
  130 
  131 -- | The total of the measure applied to the set of real numbers.
  132 total :: (Ord a, Num a) => Measure a -> a
  133 total (Measure p) =
  134     case Piecewise.toAscPieces p of
  135         [] -> 0
  136         ps -> eval (snd (last ps)) 0
  137 
  138 -- | The 'support' is the smallest closed, contiguous interval \( [x,y] \)
  139 -- outside of which the measure is zero.
  140 --
  141 -- Returns 'Nothing' if the interval is empty.
  142 support :: (Ord a, Num a) => Measure a -> Maybe (a, a)
  143 support (Measure pieces) =
  144     case Piecewise.toAscPieces pieces of
  145         [] -> Nothing
  146         ps -> Just (fst $ head ps, fst $ last ps)
  147 
  148 -- | Check whether a signed measure is positive.
  149 --
  150 -- A signed measure is /positive/ if the measure of any set
  151 -- is nonnegative. In other words a positive signed measure
  152 -- is just a measure in the ordinary sense.
  153 --
  154 -- This test is nontrivial, as we have to check that the distribution
  155 -- function is monotonically increasing.
  156 isPositive :: (Ord a, Num a, Fractional a) => Measure a -> Bool
  157 isPositive (Measure m) = go 0 $ Piecewise.toAscPieces m
  158   where
  159     go _ [] =
  160         True
  161     go before ((x, o) : []) =
  162         eval before x <= eval o x
  163     go before ((x1, o) : xos@((x2, _) : _)) =
  164         (eval before x1 <= eval o x1)
  165         && Poly.isMonotonicallyIncreasingOn o (x1,x2)
  166         && go o xos
  167 
  168 {-----------------------------------------------------------------------------
  169     Operations
  170     Numerical
  171 ------------------------------------------------------------------------------}
  172 -- | Add two measures.
  173 --
  174 -- > total (add mx my) = total mx + total my
  175 add :: (Ord a, Num a) => Measure a -> Measure a -> Measure a
  176 add (Measure mx) (Measure my) =
  177     Measure $ trim $ Piecewise.zipPointwise (+) mx my
  178 
  179 -- | Scale a measure by a constant.
  180 --
  181 -- > total (scale a mx) = a * total mx
  182 scale :: (Ord a, Num a) => a -> Measure a -> Measure a
  183 scale 0 (Measure _) = zero
  184 scale x (Measure m) = Measure $ Piecewise.mapPieces (Poly.scale x) m
  185 
  186 -- | Translate a measure along the number line.
  187 --
  188 -- > eval (distribution (translate y m)) x
  189 -- >    = eval (distribution m) (x - y)
  190 translate :: (Ord a, Num a, Fractional a) => a -> Measure a -> Measure a
  191 translate y (Measure m) =
  192     Measure $ Piecewise.translateWith Poly.translate y m

  194 {-----------------------------------------------------------------------------
  195     Operations
  196     Decomposition into continuous and discrete measures,
  197     needed for convolution.
  198 ------------------------------------------------------------------------------}
  199 -- | Measure that is absolutely continuous
  200 -- with respect to the Lebesgue measure,
  201 -- Represented via its distribution function.
  202 newtype Continuous a = Continuous { unContinuous :: Piecewise (Poly a) }
  203     -- INVARIANT: The last piece is @Poly.constant p@ for some @p :: a@.
  204 
  205 -- | Density function (Radon–Nikodym derivative) of an absolutely
  206 -- continuous measure.
  207 newtype Density a = Density (Piecewise (Poly a))
  208     -- INVARIANT: The last piece is @Poly.constant 0@.
  209 
  210 -- | Density function of an absolutely continuous measure.
  211 toDensity
  212     :: (Ord a, Num a, Fractional a)
  213     => Continuous a -> Density a
  214 toDensity = Density . Piecewise.mapPieces Poly.differentiate . unContinuous
  215 
  216 -- | Decompose a mixed measure into
  217 -- a continuous measure and a discrete measure.
  218 -- See also [Lebesgue's decomposition theorem
  219 -- ](https://en.wikipedia.org/wiki/Lebesgue%27s_decomposition_theorem)
  220 decompose
  221     :: (Ord a, Num a, Fractional a)
  222     => Measure a -> (Continuous a, D.Discrete a)
  223 decompose (Measure m) =
  224     ( Continuous $ trim $ Piecewise.fromAscPieces withoutJumps
  225     , D.fromMap $ Map.fromList jumps
  226     )
  227   where
  228     pieces = Piecewise.toAscPieces m
  229 
  230     withoutJumps =
  231         zipWith (\(x,o) j -> (x, o - Poly.constant j)) pieces totalJumps
  232     totalJumps = tail $ scanl' (+) 0 $ map snd jumps
  233 
  234     jumps = go 0 pieces
  235       where
  236         go _ [] = []
  237         go prev ((x,o) : xos) =
  238             (x, Poly.eval o x - Poly.eval prev x) : go o xos

  240 {-----------------------------------------------------------------------------
  241     Operations
  242     Convolution
  243 ------------------------------------------------------------------------------}
  244 {-$ NOTE [Convolution]
  245 
  246 In order to compute a convolution,
  247 we convolve a density with the distribution function.
  248 
  249 Let $f$ denote a density, which can be continuous or a Dirac delta.
  250 Let $G$ denote a distribution function.
  251 Let $H = f * G$ be the result of the convolution.
  252 It can be shown that this is the distribution function of the
  253 convolution of the densities, $h = f * g$.
  254 
  255 The formula for convolution is
  256 
  257 $ H(y) = ∫ f(y - x) G(x) dx = ∫ f (x) G(y - x) dx$.
  258 
  259 When $f$ is a sum of delta functions, $f = Σ w_j delta_{x_j}(x)$,
  260 this integral becomes ($y - x = x_j$ => $x = y - x_j$)
  261 
  262 $ H(y) = Σ w_j G(y - x_j) $.
  263 
  264 When $f$ is a piecewise polynomial, we can convolve the pieces.
  265 
  266 When convolving with a distribution function, the final piece
  267 will be a constant $g_n$ on the interval $[x_n,∞)$.
  268 In this case, the convolution is given by
  269 
  270 \[
  271 H(y)
  272     = ∫ f (x) G(y - x) dx
  273     = ∫_{ -∞}^{y-x_n} f(x) g_n dx
  274     = g_n F(y-x_n)
  275 \]
  276 
  277 where $F$ is the distribution function of the density $f$.
  278 -}
  279 
  280 -- | Convolve a discrete measure with a mixed measure.
  281 --
  282 -- See NOTE [Convolution].
  283 convolveDiscrete
  284     :: (Ord a, Num a, Fractional a)
  285     => D.Discrete a -> Measure a -> Measure a
  286 convolveDiscrete f gg =
  287     foldr add zero
  288         [ scale w (translate x gg)
  289         | (x, w) <- Map.toAscList $ D.toMap f
  290         ]
  291 
  292 -- | Convolve an absolutely continuous measure with a mixed measure.
  293 --
  294 -- See NOTE [Convolution].
  295 convolveContinuous
  296     :: (Ord a, Num a, Fractional a)
  297     => Continuous a -> Measure a -> Measure a
  298 convolveContinuous (Continuous ff) (Measure gg)
  299     | null ffpieces = zero
  300     | null ggpieces = zero
  301     | otherwise = Measure $ trim $ boundedConvolutions + lastConvolution
  302   where
  303     ffpieces = Piecewise.toAscPieces ff
  304     ggpieces = Piecewise.toAscPieces gg
  305 
  306     Density f = toDensity (Continuous ff)
  307     fpieces = Piecewise.toAscPieces f
  308 
  309     -- Pieces on the bounded intervals
  310     boundedPieces xos =
  311         zipWith (\(x,o) (y,_) -> (x, y, o)) xos (drop 1 xos)
  312 
  313     boundedConvolutions =
  314         sum $
  315             [ Piecewise.fromAscPieces (Poly.convolve fo ggo)
  316             | fo <- boundedPieces fpieces
  317             , ggo <- boundedPieces ggpieces
  318             ]
  319 
  320     (xlast, plast) = last ggpieces
  321     glast = case Poly.toCoefficients plast of
  322         [] -> 0
  323         (a0:_) -> a0
  324     lastConvolution =
  325         Piecewise.mapPieces (Poly.scale glast)
  326         $ Piecewise.translateWith Poly.translate xlast ff
  327 
  328 -- | Additive convolution of two measures.
  329 --
  330 -- Properties:
  331 --
  332 -- > convolve (dirac x) (dirac y) = dirac (x + y)
  333 -- >
  334 -- > convolve mx my               =  convolve my mx
  335 -- > convolve (add mx my) mz      =  add (convolve mx mz) (convolve my mz)
  336 -- > translate z (convolve mx my) =  convolve (translate z mx) my
  337 -- > total (convolve mx my)       =  total mx * total my
  338 convolve
  339     :: (Ord a, Num a, Fractional a)
  340     => Measure a -> Measure a -> Measure a
  341 convolve mx my =
  342     add (convolveContinuous contx my) (convolveDiscrete deltax my)
  343   where
  344     (contx, deltax) = decompose mx
