   48 -- | An 'Outcome' is the result of an activity that takes time,
   49 -- such as a distributed computation, communication, bus ride, … .
   50 --
   51 -- 'Outcome's can be composed in sequence or in parallel.
   52 class (Ord (Duration o), Num (Duration o)) => Outcome o where
   53     -- | Numerical type representing times in \( [0,+∞) \).
   54     --
   55     -- For example 'Double' or 'Rational'.
   56     type Duration o
   57 
   58     -- | The outcome that never finishes.
   59     never :: o
   60 
   61     -- | The outcome that succeeds after waiting for a fixed amount of time.
   62     wait :: Duration o -> o
   63 
   64     -- | Sequential composition:
   65     --
   66     -- First run the outcome on the left,
   67     -- then run the outcome on the right.
   68     sequentially :: o -> o -> o
   69 
   70     -- | Infix operator synonym for 'sequentially'.
   71     (.>>.) :: o -> o -> o
   72     (.>>.) = sequentially
   73 
   74     -- | Parallel composition, first to finish:
   75     --
   76     -- Run two outcomes in parallel,
   77     -- finish as soon as any one of them finishes.
   78     firstToFinish :: o -> o -> o
   79 
   80     -- | Infix operator synonym for 'firstToFinish'.
   81     (.\/.) :: o -> o -> o
   82     (.\/.) = firstToFinish
   83 
   84     -- | Parallel composiiton, last to finish:
   85     --
   86     -- Run two outcomes in parallel,
   87     -- finish after all of them have finished.
   88     lastToFinish :: o -> o -> o
   89 
   90     -- | Infix operator synonym for 'lastToFinish'.
   91     (./\.) :: o -> o -> o
   92     (./\.) = lastToFinish
   93 
   94 {-----------------------------------------------------------------------------
   95     Eventually
   96 ------------------------------------------------------------------------------}
   97 -- | 'Eventually' represents a value that either eventually occurs
   98 -- or is eventually abandoned.
   99 --
  100 -- Similar to the 'Maybe' type, but with a different 'Ord' instance:
  101 -- @Occurs x < Abandoned@ for all @x@.
  102 --
  103 data Eventually a
  104     = Occurs a
  105     | Abandoned
  106     deriving (Eq, Show)
  107 
  108 -- | For all @x@, we have @Occurs x < Abandoned@.
  109 instance Ord a => Ord (Eventually a) where
  110     compare Abandoned Abandoned = EQ
  111     compare Abandoned (Occurs _) = GT
  112     compare (Occurs _) Abandoned = LT
  113     compare (Occurs x) (Occurs y) = compare x y
  114 
  115 instance Functor Eventually where
  116     fmap _ Abandoned = Abandoned
  117     fmap f (Occurs x) = Occurs (f x)
  118 
  119 -- |
  120 -- > Abandoned <*> _ = Abandoned
  121 -- > _ <*> Abandoned = Abandoned
  122 instance Applicative Eventually where
  123     pure = Occurs
  124 
  125     Abandoned <*> Abandoned = Abandoned
  126     Abandoned <*> (Occurs _) = Abandoned
  127     (Occurs _) <*> Abandoned = Abandoned
  128     (Occurs f) <*> (Occurs y) = Occurs (f y)
  129 
  130 -- | Helper function to eliminate 'Eventually'.
  131 --
  132 -- See also: 'maybe'.
  133 eventually :: b -> (a -> b) -> Eventually a -> b
  134 eventually b _ Abandoned = b
  135 eventually _ f (Occurs x) = f x
  136 
  137 -- | Helper function that converts 'Maybe' to 'Eventually'.
  138 eventuallyFromMaybe :: Maybe a -> Eventually a
  139 eventuallyFromMaybe Nothing = Abandoned
  140 eventuallyFromMaybe (Just x) = Occurs x
  141 
  142 -- | Helper function that converts 'Eventually' to 'Maybe'.
  143 maybeFromEventually :: Eventually a -> Maybe a
  144 maybeFromEventually Abandoned = Nothing
  145 maybeFromEventually (Occurs x) = Just x

  151 -- | 'DeltaQ' — quality attenuation.
  152 --
  153 -- 'DeltaQ' is a probability distribution of time.
  154 --
  155 -- Specifically, 'DeltaQ' is the probability distribution
  156 -- of finish times for an outcome.
  157 class   ( Eq (Probability o)
  158         , Enum (Probability o)
  159         , Num (Probability o)
  160         , Fractional (Probability o)
  161         , Outcome o
  162         )
  163     => DeltaQ o
  164   where
  165     -- | Numerical type representing probabilities in \( [0,1] \).
  166     --
  167     -- For example 'Double' or 'Rational'.
  168     type Probability o
  169 
  170     -- | Left-biased random choice.
  171     --
  172     -- @choice p@ chooses the left argument with probablity @p@
  173     -- and the right argument with probability @(1-p)@.
  174     choice :: Probability o -> o -> o -> o
  175 
  176     -- | Random choice between multiple alternatives
  177     --
  178     -- @choices [(w_1, o_1), (w_2, o_2), …]@ chooses randomly between multiple
  179     -- outcomes. The probability @p_i@ for choosing the outcome @o_i@ is
  180     -- determined by the weights as @p_i = w_i / (w_1 + w_2 + …)@.
  181     choices :: [(Probability o, o)] -> o
  182     choices [] = never
  183     choices wos =
  184         foldr (uncurry choice) never
  185         $ zipWith (\wtot (w, o) -> (w / wtot, o)) ws wos
  186       where
  187         ws = scanr1 (+) (map fst wos)
  188 
  189     -- | Uniform probability distribution on a time interval.
  190     uniform :: Duration o -> Duration o -> o
  191 
  192     -- | Probability of /not/ finishing.
  193     failure :: o -> Probability o
  194 
  195     -- | Probability of finishing within the given time @t@.
  196     --
  197     -- \"Within\" is inclusive,
  198     -- i.e. this returns the probability that the finishing time is @<= t@.
  199     successWithin :: o -> Duration o -> Probability o
  200 
  201     -- | Given a probability @p@, return the smallest time @t@
  202     -- such that the probability of completing within that time
  203     -- is at least @p@.
  204     --
  205     -- Return 'Abandoned' if the given probability
  206     -- exceeds the probability of finishing.
  207     quantile :: Probability o -> o -> Eventually (Duration o)
  208 
  209     -- | The earliest finish time with non-zero probability.
  210     --
  211     -- Return 'Abandoned' if the outcome is 'never'.
  212     earliest :: o -> Eventually (Duration o)
  213 
  214     -- | The last finish time which still has non-zero probability to occur.
  215     --
  216     -- Return 'Abandoned' if arbitrarily late times are possible.
  217     deadline :: o -> Eventually (Duration o)
  218

  231 {-$properties-outcome
  232 
  233 'never'
  234 
  235 > never .>>. y = never
  236 > never ./\. y = never
  237 > never .\/. y = y
  238 >
  239 > x .>>. never = never
  240 > x ./\. never = never
  241 > x .\/. never = x

  243 'wait'
  244 
  245 > wait t .>>. wait s  =  wait (t+s)
  246 > wait t ./\. wait s  =  wait (max t s)
  247 > wait t .\/. wait s  =  wait (min t s)
  248 
  249 '(.>>.)'
  250 
  251 > (x .>>. y) .>>. z  =  x .>>. (y .>>. z)
  252 
  253 '(./\.)'
  254 
  255 > (x ./\. y) ./\. z  =  x ./\. (y ./\. z)
  256 >
  257 > x ./\. y  =  y ./\. x

  261 > (x .\/. y) .\/. z  =  x .\/. (y .\/. z)

  267 {-$properties-deltaq
  268 
  269 'choice'
  270 
  271 > choice 1 x y = x
  272 > choice 0 x y = y
  273 >
  274 > choice p x y .>>. z  =  choice p (x .>>. z) (y .>>. z)
  275 > choice p x y ./\. z  =  choice p (x ./\. z) (y ./\. z)
  276 > choice p x y .\/. z  =  choice p (x .\/. z) (y .\/. z)
  277 
  278 'choices'
  279 
  280 > choices [] = never
  281 > choices ((w,o) : wos) = choice p o (choices wos)
  282 >   where  p = w / (w + sum (map fst wos))
  283 
  284 'uniform'
  285 
  286 > wait t .>>. uniform r s  =  uniform (t+r) (t+s)
  287 > uniform r s .>>. wait t  =  uniform (r+t) (s+t)
  288 
  289 'failure'
  290 
  291 > failure never      = 1
  292 > failure (wait t)   = 0
  293 > failure (x .>>. y) = 1 - (1 - failure x) * (1 - failure y)
  294 > failure (x ./\. y) = 1 - (1 - failure x) * (1 - failure y)
  295 > failure (x .\/. y) = failure x * failure y
  296 >
  297 > failure (choice p x y) = p * failure x + (1-p) * failure y
  298 > failure (uniform r s)  = 0
  299 
  300 'successWithin'
  301 
  302 > successWithin never    t = 0
  303 > successWithin (wait s) t = if t < s then 0 else 1
  304 >
  305 > successWithin (x ./\. y) t =
  306 >   successWithin t x * successWithin t y
  307 > successWithin (x .\/. y) t =
  308 >   1 - (1 - successWithin t x) * (1 - successWithin t y)
  309 >
  310 > successWithin (choice p x y) t =
  311 >   p * successWithin t x + (1-p) * successWithin t y
  312 > successWithin (uniform r s) t
  313 >   | t < r           = 0
  314 >   | r <= t && t < s = (t-r) / (s-r)
  315 >   | s <= t          = 1
  316 
  317 'quantile'
  318 
  319 > p <= q  implies  quantile p o <= quantile q o
  320 >
  321 > quantile p never    = Abandoned
  322 > quantile p (wait t) = if p > 0 then Occurs t else Abandoned
  323 >
  324 > quantile p (uniform r s)  =  r + p*(s-t)  if r <= s
  325 
  326 'earliest'
  327 
  328 > earliest never      = Abandoned
  329 > earliest (wait t)   = Occurs t
  330 > earliest (x .>>. y) = (+) <$> earliest x <*> earliest y
  331 > earliest (x ./\. y) = max (earliest x) (earliest y)
  332 > earliest (x .\/. y) = min (earliest x) (earliest y)
  333 >
  334 > earliest (choice p x y) = min (earliest x) (earliest y)  if p ≠ 0, p ≠ 1
  335 > earliest (uniform r s)  = Occurs r   if r <= s
  336 
  337 'deadline'
  338 
  339 > deadline never      =  Abandoned
  340 > deadline (wait t)   =  Occurs t
  341 > deadline (x .>>. y) =  (+) <$> deadline x <*> deadline y
  342 > deadline (x ./\. y) =  max (deadline x) (deadline y)
  343 >
  344 > deadline (x .\/. y) =  min (deadline x) (deadline y)
  345 >   if failure x = 0, failure y = 0
  346 >
  347 > deadline (choice p x y) = max (deadline x) (deadline y)
  348 >   if p ≠ 0, p ≠ 1, failure x = 0, failure y = 0
  349 >
  350 >
  351 > deadline (uniform r s)  = Occurs s   if r <= s
  352 
  353 -}
