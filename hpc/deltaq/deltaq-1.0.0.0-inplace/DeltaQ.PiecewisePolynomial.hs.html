   52 {-----------------------------------------------------------------------------

   55 -- | Probability distribution of durations.
   56 newtype DQ = DQ (Measure Rational)
   57     deriving (Eq, Show, NFData)
   58 
   59 -- | Get the distribution function as piecewise function of polynomials.
   60 distribution :: DQ -> Piecewise (Poly Rational)
   61 distribution (DQ m) = Measure.distribution m
   62 
   63 -- | Interpret a finite, signed 'Measure' as a probability distribution.
   64 --
   65 -- In order to admit an interpretation as probability, the measure needs
   66 -- to be positive, if that is not the case, the function returns 'Nothing'.
   67 --
   68 -- Note: Due to rare numerical issues,
   69 -- the check 'Measure.isPositive' can be false negative.
   70 -- Work around using 'unsafeFromPositiveMeasure'.
   71 fromPositiveMeasure :: Measure Rational -> Maybe DQ
   72 fromPositiveMeasure m
   73     | Measure.isPositive m = Just (unsafeFromPositiveMeasure m)
   74     | otherwise = Nothing
   75 
   76 -- | Interpret a finite, positive 'Measure' as a probability distribution.
   77 --
   78 -- /The precondition that the measure satisfies 'Measure.isPositive'
   79 -- is not checked!/
   80 unsafeFromPositiveMeasure :: Measure Rational -> DQ
   81 unsafeFromPositiveMeasure = DQ
   82 
   83 -- | Helper function for lifting a binary operation on distribution functions.
   84 onDistribution2
   85     :: (a ~ Rational)
   86     => String
   87     -> (Piecewise (Poly a) -> Piecewise (Poly a) -> Piecewise (Poly a))
   88     -> DQ -> DQ -> DQ
   89 onDistribution2 err f (DQ mx) (DQ my) =
   90     DQ
   91     $ fromMaybe impossible
   92     $ Measure.fromDistribution
   93     $ f (Measure.distribution mx) (Measure.distribution my)
   94   where
   95     impossible = error $ "impossible: not a finite measure in " <> err
   96 
   97 -- | Size of the representation of a probability distribution,
   98 -- i.e. number of pieces of the piecewise function and degrees
   99 -- of the polynomials.
  100 --
  101 -- This quantity is relevant to stating and analyzing
  102 -- the asymptotic time complexity of operations.
  103 complexity :: DQ -> Int
  104 complexity (DQ m) = sum (map complexityOfPiece pieces)
  105   where
  106     pieces = Piecewise.toAscPieces $ Measure.distribution m
  107     complexityOfPiece = (+1) . max 0 . Poly.degree . snd
  108 
  109 {-----------------------------------------------------------------------------
  110     Operations
  111 ------------------------------------------------------------------------------}
  112 instance Outcome DQ where
  113     type Duration DQ = Rational
  114 
  115     never = DQ Measure.zero
  116 
  117     wait t = DQ $ Measure.dirac t
  118 
  119     sequentially (DQ mx) (DQ my) = DQ (Measure.convolve mx my)
  120 
  121     firstToFinish = onDistribution2 "firstToFinish" $ \x y -> x + y - x * y
  122 
  123     lastToFinish = onDistribution2 "lastToFinish" (*)
  124 
  125 instance DeltaQ DQ where
  126     type Probability DQ = Rational
  127 
  128     choice p = onDistribution2 "choice" $ \x y ->
  129         scale p x + scale (1 - p) y
  130       where
  131         scale = Piecewise.mapPieces . Poly.scale
  132 
  133     uniform a = DQ . Measure.uniform a
  134 
  135     successWithin (DQ m) = Function.eval (Measure.distribution m)
  136 
  137     failure (DQ m) = 1 - Measure.total m
  138 
  139     quantile p (DQ m) =
  140         eventuallyFromMaybe
  141         $ quantileFromMonotone (Measure.distribution m) p
  142 
  143     earliest (DQ m) = eventuallyFromMaybe $ fmap fst $ Measure.support m
  144 
  145     deadline (DQ m)= eventuallyFromMaybe $ fmap snd $ Measure.support m

  151 -- | Helper type for segements of a piecewise functions.
  152 data Segment a b
  153     = Jump a (b,b)
  154     | Polynomial (a,a) (b,b) (Poly a)
  155     | End a b
  156 
  157 -- | Helper function that elaborates a piecewise function
  158 -- into a list of segments.
  159 toSegments :: (a ~ Rational) => Piecewise (Poly a) -> [Segment a a]
  160 toSegments = goJump 0 . Piecewise.toAscPieces
  161   where
  162     goJump _ [] = []
  163     goJump prev ((x1, o) : xos)
  164         | y1 - y0 > 0 = Jump x1 (y0, y1) : nexts
  165         | otherwise = nexts
  166       where
  167         y1 = Poly.eval o x1
  168         y0 = Poly.eval prev x1
  169         nexts = goPoly x1 y1 o xos
  170 
  171     goPoly x1 y1 o [] =
  172         End x1 y1 : goJump o []
  173     goPoly x1 y1 o xos@((x2, _) : _) =
  174         Polynomial (x1, x2) (y1, Poly.eval o x2) o : goJump o xos
  175         -- TODO: What about the case where y1 == y2, i.e. a constant Polynomial?
  176 
  177 -- | Compute a quantile from a monotonically increasing function.
  178 quantileFromMonotone :: (a ~ Rational) => Piecewise (Poly a) -> a -> Maybe a
  179 quantileFromMonotone pieces = findInSegments segments
  180   where
  181     segments = toSegments pieces
  182 
  183     findInSegments _ 0
  184         = Just 0
  185     findInSegments [] _
  186         = Nothing
  187     findInSegments (Jump x1 (y1, y2) : xys) y
  188         | y1 < y && y <= y2 = Just x1
  189         | otherwise = findInSegments xys y
  190     findInSegments (Polynomial (x1, x2) (y1, y2) o : xys) y
  191         | y1 < y && y <= y2 = Poly.root precision y (x1, x2) o
  192         | otherwise = findInSegments xys y
  193     findInSegments (End x1 y1 : _) y
  194         | y1 == y = Just x1
  195         | otherwise = Nothing
  196 
  197 precision :: Rational
  198 precision = 1 / 10^(10 :: Integer)
