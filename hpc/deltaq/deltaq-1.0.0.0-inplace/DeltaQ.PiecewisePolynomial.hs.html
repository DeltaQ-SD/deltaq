   66 -- | Probability distribution of durations.
   67 newtype DQ = DQ (Measure Rational)
   68     deriving (Eq, Show, NFData)
   69 
   70 -- | Get the distribution function as piecewise function of polynomials.
   71 distribution :: DQ -> Piecewise (Poly Rational)
   72 distribution (DQ m) = Measure.distribution m
   73 
   74 -- | Interpret a finite, signed 'Measure' as a probability distribution.
   75 --
   76 -- In order to admit an interpretation as probability, the measure needs
   77 -- to be positive, if that is not the case, the function returns 'Nothing'.
   78 --
   79 -- Note: Due to rare numerical issues,
   80 -- the check 'Measure.isPositive' can be false negative.
   81 -- Work around using 'unsafeFromPositiveMeasure'.
   82 fromPositiveMeasure :: Measure Rational -> Maybe DQ
   83 fromPositiveMeasure m
   84     | Measure.isPositive m = Just (unsafeFromPositiveMeasure m)
   85     | otherwise = Nothing
   86 
   87 -- | Interpret a finite, positive 'Measure' as a probability distribution.
   88 --
   89 -- /The precondition that the measure satisfies 'Measure.isPositive'
   90 -- is not checked!/
   91 unsafeFromPositiveMeasure :: Measure Rational -> DQ
   92 unsafeFromPositiveMeasure = DQ
   93 
   94 -- | Helper function for lifting a binary operation on distribution functions.
   95 onDistribution2
   96     :: (a ~ Rational)
   97     => String
   98     -> (Piecewise (Poly a) -> Piecewise (Poly a) -> Piecewise (Poly a))
   99     -> DQ -> DQ -> DQ
  100 onDistribution2 err f (DQ mx) (DQ my) =
  101     DQ
  102     $ fromMaybe impossible
  103     $ Measure.fromDistribution
  104     $ f (Measure.distribution mx) (Measure.distribution my)
  105   where
  106     impossible = error $ "impossible: not a finite measure in " <> err
  107 
  108 -- | Size of the representation of a probability distribution,
  109 -- i.e. number of pieces of the piecewise function and degrees
  110 -- of the polynomials.
  111 --
  112 -- This quantity is relevant to stating and analyzing
  113 -- the asymptotic time complexity of operations.
  114 complexity :: DQ -> Int
  115 complexity (DQ m) = sum (map complexityOfPiece pieces)
  116   where
  117     pieces = Piecewise.toAscPieces $ Measure.distribution m
  118     complexityOfPiece = (+1) . max 0 . Poly.degree . snd

  123 instance Outcome DQ where
  124     type Duration DQ = Rational
  125 
  126     never = DQ Measure.zero
  127 
  128     wait t = DQ $ Measure.dirac t
  129 
  130     sequentially (DQ mx) (DQ my) = DQ (Measure.convolve mx my)
  131 
  132     firstToFinish = onDistribution2 "firstToFinish" $ \x y -> x + y - x * y
  133 
  134     lastToFinish = onDistribution2 "lastToFinish" (*)

  136 instance DeltaQ DQ where
  137     type Probability DQ = Rational
  138 
  139     choice p = onDistribution2 "choice" $ \x y ->
  140         scale p x + scale (1 - p) y
  141       where
  142         scale = Piecewise.mapPieces . Poly.scale
  143 
  144     uniform a = DQ . Measure.uniform a
  145 
  146     successWithin (DQ m) = Function.eval (Measure.distribution m)
  147 
  148     failure (DQ m) = 1 - Measure.total m
  149 
  150     quantile p (DQ m) =
  151         eventuallyFromMaybe
  152         $ quantileFromMonotone (Measure.distribution m) p
  153 
  154     earliest (DQ m) = eventuallyFromMaybe $ fmap fst $ Measure.support m
  155 
  156     deadline (DQ m)= eventuallyFromMaybe $ fmap snd $ Measure.support m

  162 -- | Helper type for segements of a piecewise functions.
  163 data Segment a b
  164     = Jump a (b,b)
  165     | Polynomial (a,a) (b,b) (Poly a)
  166     | End a b
  167     deriving (Eq, Show)
  168 
  169 -- | Helper function that elaborates a piecewise function
  170 -- into a list of segments.
  171 toSegments :: (a ~ Rational) => Piecewise (Poly a) -> [Segment a a]
  172 toSegments = goJump 0 . Piecewise.toAscPieces
  173   where
  174     goJump _ [] = []
  175     goJump prev ((x1, o) : xos)
  176         | y1 - y0 > 0 = Jump x1 (y0, y1) : nexts
  177         | otherwise = nexts
  178       where
  179         y1 = Poly.eval o x1
  180         y0 = Poly.eval prev x1
  181         nexts = goPoly x1 y1 o xos
  182 
  183     goPoly x1 y1 o [] =
  184         End x1 y1 : goJump o []
  185     goPoly x1 y1 o xos@((x2, _) : _) =
  186         Polynomial (x1, x2) (y1, Poly.eval o x2) o : goJump o xos
  187         -- TODO: What about the case where y1 == y2, i.e. a constant Polynomial?
  188 
  189 -- | Compute a quantile from a monotonically increasing function.
  190 quantileFromMonotone :: (a ~ Rational) => Piecewise (Poly a) -> a -> Maybe a
  191 quantileFromMonotone pieces = findInSegments segments
  192   where
  193     segments = toSegments pieces
  194 
  195     findInSegments _ 0
  196         = Just 0
  197     findInSegments [] _
  198         = Nothing
  199     findInSegments (Jump x1 (y1, y2) : xys) y
  200         | y1 < y && y <= y2 = Just x1
  201         | otherwise = findInSegments xys y
  202     findInSegments (Polynomial (x1, x2) (y1, y2) o : xys) y
  203         | y1 < y && y <= y2 = Poly.root precision y (x1, x2) o
  204         | otherwise = findInSegments xys y
  205     findInSegments (End x1 y1 : _) y
  206         | y1 == y = Just x1
  207         | otherwise = Nothing
  208 
  209 precision :: Rational
  210 precision = 1 / 10^(10 :: Integer)

  216 -- | Test whether the given probability distribution of completion times
  217 -- is equal to or better than a given
  218 -- __quantitative timeliness agreement__ (QTA).
  219 --
  220 -- > p `meetsQTA` qta = True
  221 --
  222 -- if and only
  223 --
  224 -- > âˆ€ t. successWithin o t >= successWithin qta t
  225 meetsQTA :: DQ -> DQ -> Bool
  226 meetsQTA m1 m2 =
  227     all isNonNegative $ toSegments $ distribution m1 - distribution m2
  228   where
  229     isNonNegative (Jump _ (y1, y2)) =
  230         y1 >= 0 && y2 >= 0
  231     isNonNegative (Polynomial (x1, x2) _ poly) =
  232         let compareToZero = Poly.compareToZero (x1, x2, poly)
  233         in  compareToZero == Just GT || compareToZero == Just EQ
  234     isNonNegative (End _ y) =
  235         y >= 0

  241 -- | Compute the success probability of a 'DQ',
  242 -- and the first commonly used 'Moments' of the
  243 -- probability distribution conditioned on success.
  244 moments :: DQ -> (Rational, Moments Rational)
  245 moments (DQ m)
  246     | success == 0 =
  247         (0, Moments{mean = 0, variance = 0, skewness = 0, kurtosis = 1})
  248     | otherwise =
  249         (success, Prob.moments conditional)
  250   where
  251     success = Measure.total m
  252     conditional = Prob.unsafeFromMeasure $ Measure.scale (1/success) m
