   33 -- | The \"slack or hazard\" represents the distance between
   34 -- a reference point in (time, probability) space
   35 -- and a given 'DeltaQ'.
   36 --
   37 -- * 'Slack' represents the case where the 'DeltaQ' __meets__
   38 --   the performance requirements set by the reference point.
   39 -- * 'Hazard' represents the case where the 'DeltaQ' __fails__ to meet
   40 --   the performance requirements set by the reference point.
   41 --
   42 -- Both cases include information of how far the reference point is
   43 -- away.
   44 data SlackOrHazard o
   45     = Slack (Duration o) (Probability o)
   46     -- ^ We have some slack.
   47     -- Specifically, we have 'Duration' at the same probability as the reference,
   48     -- and 'Probability' at the same duration as the reference.
   49     | Hazard (Eventually (Duration o)) (Probability o)
   50     -- ^ We fail to meet the reference point.
   51     -- Specifically,
   52     -- we overshoot by 'Duration' at the same probability as the reference,
   53     -- and by 'Probability' at the same duration as the reference.
   54 
   55 deriving instance (Eq (Duration o), Eq (Probability o))
   56     => Eq (SlackOrHazard o)
   57 
   58 deriving instance (Show (Duration o), Show (Probability o))
   59     => Show (SlackOrHazard o)
   60 
   61 -- | Test whether the given 'SlackOrHazard' is 'Slack'.
   62 isSlack :: SlackOrHazard o -> Bool
   63 isSlack (Slack _ _) = True
   64 isSlack _ = False
   65 
   66 -- | Test whether the given 'SlackOrHazard' is 'Hazard'.
   67 isHazard :: SlackOrHazard o -> Bool
   68 isHazard (Hazard _ _) = True
   69 isHazard _ = False
   70 
   71 -- | Compute \"slack or hazard\" with respect to a given reference point.
   72 meetsRequirement
   73     :: DeltaQ o => o -> (Duration o, Probability o) -> SlackOrHazard o
   74 meetsRequirement o (t,p)
   75     | dp >= 0 = Slack dt dp
   76     | Abandoned <- t' = Hazard Abandoned (negate dp)
   77     | otherwise = Hazard (Occurs $ negate dt) (negate dp)
   78   where
   79     dp = p' - p
   80     dt = t - eventually err id t'
   81 
   82     t' = quantile o p
   83     p' = o `successWithin` t
   84 
   85     err = error "distanceToReference: inconsistency"
