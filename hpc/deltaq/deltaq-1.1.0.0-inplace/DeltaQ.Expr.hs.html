   69 {-----------------------------------------------------------------------------
   70     O

   72 -- | Outcome expression.
   73 --
   74 -- This type includes graphical annotations for the rendering
   75 -- as outcome diagram.
   76 newtype O = O { unO :: Term String }
   77     -- INVARIANT: Terms satisfy 'isNormalizedAssoc'.
   78     deriving (Show, Generic, NFData)
   79 
   80 -- | Outcome variable, given by a unique name.
   81 var :: String -> O
   82 var = O . Var
   83 
   84 -- | Observation location, with a label.
   85 --
   86 -- Rendered as a box with a label in the outcome diagram.
   87 -- Put this in sequence with other outcome expressions to highlight
   88 -- their observation locations.
   89 loc :: String -> O
   90 loc = O . Loc
   91 
   92 -- | Substitute all variables by outcome expressions.
   93 substitute
   94     :: (String -> O)
   95         -- ^ Assignment of variable names to outcome expressions.
   96     -> O
   97         -- ^ Outcome expression in which we substitute.
   98     -> O
   99 substitute f (O term) = O $ normalize1Assoc $ term >>= unO . f
  100 
  101 -- | Map an outcome expression 'O' to
  102 -- probability distributions of completion times 'DQ',
  103 -- given a basic mapping from 'var' names to completion times.
  104 toDeltaQ :: (String -> DQ) -> O -> DQ
  105 toDeltaQ f (O term) = go term
  106   where
  107     go (Var v)  = f v
  108     go Never    = never
  109     go Wait0    = wait 0
  110     go (Wait t) = wait t
  111     go (Loc  _) = wait 0
  112     go (Seq   xs) = foldr1 (.>>.) $ map go xs
  113     go (Last  xs) = foldr1 (./\.) $ map go xs
  114     go (First xs) = foldr1 (.\/.) $ map go xs
  115     go (Choices wxs) = choices [ (w, go x) | (w, x) <- wxs ]
  116 
  117 -- | Outcome expressions can be combined sequentially and in parallel.
  118 instance Outcome O where
  119     type Duration O = Rational
  120 
  121     never = O Never
  122     wait = O . Wait
  123     sequentially  (O x) (O y) = O . normalize1Assoc $ Seq [x,y]
  124     firstToFinish (O x) (O y) = O . normalize1Assoc $ First [x,y]
  125     lastToFinish  (O x) (O y) = O . normalize1Assoc $ Last [x,y]
  126 
  127 -- | Outcome expressions can be combined with random choice.
  128 instance ProbabilisticOutcome O where
  129     type Probability O = Rational
  130 
  131     choice p (O x) (O y) = O $ Choices [(p, x), (1-p, y)]
  132     choices wos = O $ Choices [ (w, x) | (w, O x) <- wos ]

  137 -- | Term representation for outcome expressions.
  138 --
  139 -- Different terms may represent equal outcomes.
  140 data Term v
  141     = Var v
  142         -- ^ Variable, to be substituted by a more concrete outcome.
  143     | Never
  144         -- ^ Outcome that never finishes.
  145     | Wait0
  146         -- ^ Succeed immediately. Equivalent to @Wait 0@,
  147         -- but with a straight line as graphical representation.
  148     | Wait Rational
  149         -- ^ Succeed after waiting for a fixed amount of time.
  150     | Loc String
  151         -- ^ Outcome location with a label.
  152         -- Equivalent to @Wait 0@, but with a labeled box
  153         -- as graphical representation.
  154     | Seq [Term v]
  155         -- ^ Sequential composition.
  156     | Last [Term v]
  157         -- ^ Parallel composition, last to finish.
  158     | First [Term v]
  159         -- ^ Parallel composiiton, first to finish.
  160     | Choices [(Rational, Term v)]
  161         -- ^ Probabilistic choice.
  162         -- The probabilities are proportional to the given weights.
  163     deriving (Show, Eq, Ord, Generic, Functor, NFData)
  164 
  165 instance Applicative Term where
  166     pure = Var
  167     (<*>) = ap
  168 
  169 -- | '(>>=)' is substitution.
  170 instance Monad Term where
  171     m >>= g = go m
  172       where
  173         go (Var v)    = g v
  174         go Never      = Never
  175         go Wait0      = Wait0
  176         go (Wait t)   = Wait t
  177         go (Loc  s)   = Loc s
  178         go (Seq   xs) = Seq   $ map go xs
  179         go (Last  xs) = Last  $ map go xs
  180         go (First xs) = First $ map go xs
  181         go (Choices wxs) = Choices [ (w, go x) | (w, x) <- wxs ]
  182 
  183 -- | Inspect an outcome expression 'O' through its 'Term' representation.
  184 --
  185 -- The result 'Term' satisfies 'isNormalizedAssoc'.
  186 termFromOutcome :: O -> Term String
  187 termFromOutcome (O term) = term
  188 
  189 -- | Construct an outcome expression 'O' from an outcome 'Term'.
  190 outcomeFromTerm :: Term String -> O
  191 outcomeFromTerm = O . normalizeAssoc
  192 
  193 -- | Predicate that defines the \"associative normal form\" for a 'Term'.
  194 --
  195 -- Specifically, a 'Term' is said to be in \"associative normal form\"
  196 -- if the arguments to the constructors 'Seq', 'Last', and 'First'
  197 --
  198 -- * are nonempty lists, and
  199 -- * their list elements do not have an outermost constructor of the same
  200 --   kind, i.e. @Seq [Seq …, First …]@ is not allowed because
  201 --   one of the list elements for a 'Seq' constructor
  202 --   is also a 'Seq' constructor.
  203 --
  204 -- and if the argument to the constructor 'Choices'
  205 --
  206 -- * is a nonempty list
  207 --
  208 isNormalizedAssoc ::Term v -> Bool 
  209 isNormalizedAssoc (Seq xs) =
  210     not (null xs)
  211     && all (not . isSeq) xs
  212     && all isNormalizedAssoc xs
  213 isNormalizedAssoc (Last xs) =
  214     not (null xs)
  215     && all (not . isLast) xs
  216     && all isNormalizedAssoc xs
  217 isNormalizedAssoc (First xs) =
  218     not (null xs)
  219     && all (not . isFirst) xs
  220     && all isNormalizedAssoc xs
  221 isNormalizedAssoc (Choices wxs) =
  222     not (null wxs)
  223     && all isNormalizedAssoc (map snd wxs)
  224 isNormalizedAssoc _ =
  225     True
  226 
  227 -- | Check whether a 'Term' is a 'Var'.
  228 isVar :: Term v -> Bool
  229 isVar (Var _) = True
  230 isVar _       = False
  231 
  232 -- | Check whether a 'Term' is a 'Loc'.
  233 isLoc :: Term v -> Bool
  234 isLoc (Loc _) = True
  235 isLoc _       = False
  236 
  237 -- | Check whether a 'Term' is a 'Seq'.
  238 isSeq :: Term v -> Bool
  239 isSeq (Seq _) = True
  240 isSeq _ = False
  241 
  242 -- | Check whether a 'Term' is a 'Last'.
  243 isLast :: Term v -> Bool
  244 isLast (Last _) = True
  245 isLast _ = False
  246 
  247 -- | Check whether a 'Term' is a 'First'.
  248 isFirst :: Term v -> Bool
  249 isFirst (First _) = True
  250 isFirst _ = False
  251 
  252 -- | Check whether a 'Term' is a 'Choices'.
  253 isChoices :: Term v -> Bool
  254 isChoices (Choices _) = True
  255 isChoices _ = False
  256 
  257 -- | Check whether a 'Term' is a parallel operation,
  258 -- i.e. 'Last' or 'First'.
  259 isParallel :: Term v -> Bool
  260 isParallel (First ts) = not (null ts)
  261 isParallel (Last  ts) = not (null ts)
  262 isParallel _          = False
  263 
  264 -- | Maximal number of outcomes that run in parallel.
  265 --
  266 -- * The arguments of 'First' and 'Last' run in parallel.
  267 -- * The arguments of 'Seq' and 'Choices' do __not__ run in parallel.
  268 maxParallel :: Term v -> Int
  269 maxParallel (Seq   ts)    = maximum $ map maxParallel ts
  270 maxParallel (Last  ts)    = sum $ map maxParallel ts
  271 maxParallel (First ts)    = sum $ map maxParallel ts
  272 maxParallel (Choices wts) = maximum $ map (maxParallel . snd) wts
  273 maxParallel _             = 1
  274 
  275 -- | Normalize a term to \"associative normal form\".
  276 normalizeAssoc :: Term v -> Term v
  277 normalizeAssoc = everywhere normalize1Assoc
  278 
  279 -- | Apply a transformation everywhere; bottom-up.
  280 --
  281 -- See also [Scrap your boilerplate
  282 -- ](https://www.microsoft.com/en-us/research/wp-content/uploads/2003/01/hmap.pdf)
  283 -- .
  284 everywhere :: (Term v -> Term v) -> Term v -> Term v
  285 everywhere f = every
  286   where
  287     every = f . recurse
  288 
  289     recurse a@(Var _) = a
  290     recurse a@Never = a
  291     recurse a@Wait0 = a
  292     recurse a@(Wait _) = a
  293     recurse a@(Loc  _) = a
  294     recurse (Seq   xs) = Seq $ map every xs
  295     recurse (Last  xs) = Last $ map every xs
  296     recurse (First xs) = First $ map every xs
  297     recurse (Choices wxs) = Choices $ [ (w, every x) | (w, x) <- wxs]
  298 
  299 -- | Summarize all nodes; top-down, left-to-right.
  300 --
  301 -- See also [Scrap your boilerplate
  302 -- ](https://www.microsoft.com/en-us/research/wp-content/uploads/2003/01/hmap.pdf)
  303 -- .
  304 everything :: (r -> r -> r) -> (Term v -> r) -> (Term v -> r)
  305 everything combine f = recurse
  306   where
  307     recurse x@(Var _)  = f x
  308     recurse x@Never    = f x
  309     recurse x@Wait0    = f x
  310     recurse x@(Wait _) = f x
  311     recurse x@(Loc  _) = f x
  312     recurse x@(Seq   xs) =
  313         foldl' combine (f x) $ map recurse xs
  314     recurse x@(Last  xs) =
  315         foldl' combine (f x) $ map recurse xs
  316     recurse x@(First xs) =
  317         foldl' combine (f x) $ map recurse xs
  318     recurse x@(Choices wxs) =
  319         foldl' combine (f x) $ [ recurse y | (_, y) <- wxs]
  320 
  321 -- | Normalize a term to \"associative normal form\"
  322 -- under the assumptions
  323 -- that the arguments to the outermost constructor are already normalized.
  324 normalize1Assoc :: Term v -> Term v
  325 normalize1Assoc = id
  326     . normalizeEmpty
  327     . normalize1Assoc'
  328 
  329 -- | Ensure that the lists in 'Seq', 'Last', 'First'
  330 -- do not contain a term with the same constructor.
  331 normalizeEmpty :: Term v -> Term v
  332 normalizeEmpty (Seq   []) = Wait0
  333 normalizeEmpty (Last  []) = Wait0
  334 normalizeEmpty (First []) = Wait0
  335 normalizeEmpty (Choices []) = Wait0
  336 normalizeEmpty x = x
  337 
  338 -- | Ensure that the lists in 'Seq', 'Last', 'First'
  339 -- do not contain a term with the same constructor.
  340 normalize1Assoc' :: Term v -> Term v
  341 normalize1Assoc' (Seq xs) = Seq (concatMap f xs)
  342   where
  343     f Wait0 = []
  344     f (Seq ys) = ys
  345     f o = [o]
  346 normalize1Assoc' (Last xs) = Last (concatMap f xs)
  347   where
  348     f Wait0 = []
  349     f (Last ys) = ys
  350     f o = [o]
  351 normalize1Assoc' (First xs) = First (concatMap f xs)
  352   where
  353     f Wait0 = []
  354     f (First ys) = ys
  355     f o = [o]
  356 normalize1Assoc' o = o
