   69 -- | Probability distribution of durations.
   70 newtype DQ = DQ (Measure Rational)
   71     deriving (Eq, Show, NFData)
   72 
   73 -- | Get the distribution function as piecewise function of polynomials.
   74 distribution :: DQ -> Piecewise (Poly Rational)
   75 distribution (DQ m) = Measure.distribution m
   76 
   77 -- | Interpret a finite, signed 'Measure' as a probability distribution.
   78 --
   79 -- In order to admit an interpretation as probability, the measure needs
   80 -- to be positive.
   81 -- This condition is checked, and if it does not hold,
   82 -- the function returns 'Nothing'.
   83 fromPositiveMeasure :: Measure Rational -> Maybe DQ
   84 fromPositiveMeasure m
   85     | Measure.isPositive m = Just (unsafeFromPositiveMeasure m)
   86     | otherwise = Nothing
   87 
   88 -- | Interpret a finite, positive 'Measure' as a probability distribution.
   89 --
   90 -- /The precondition that the measure is positive is not checked!/
   91 unsafeFromPositiveMeasure :: Measure Rational -> DQ
   92 unsafeFromPositiveMeasure = DQ

   94 -- | Helper function for lifting a binary operation on distribution functions.
   95 onDistribution2
   96     :: (a ~ Rational)
   97     => String
   98     -> (Piecewise (Poly a) -> Piecewise (Poly a) -> Piecewise (Poly a))
   99     -> DQ -> DQ -> DQ
  100 onDistribution2 err f (DQ mx) (DQ my) =
  101     DQ
  102     $ fromMaybe impossible
  103     $ Measure.fromDistribution
  104     $ f (Measure.distribution mx) (Measure.distribution my)
  105   where
  106     impossible = error $ "impossible: not a finite measure in " <> err
  107 
  108 -- | Size of the representation of a probability distribution,
  109 -- i.e. number of pieces of the piecewise function and degrees
  110 -- of the polynomials.
  111 --
  112 -- This quantity is relevant to stating and analyzing
  113 -- the asymptotic time complexity of operations.
  114 complexity :: DQ -> Int
  115 complexity (DQ m) = sum (map complexityOfPiece pieces)
  116   where
  117     pieces = Piecewise.toAscPieces $ Measure.distribution m
  118     complexityOfPiece = (+1) . max 0 . Poly.degree . snd

  123 instance Outcome DQ where
  124     type Duration DQ = Rational
  125 
  126     never = DQ Measure.zero
  127 
  128     wait t = DQ $ Measure.dirac t
  129 
  130     sequentially (DQ mx) (DQ my) = DQ (Measure.convolve mx my)
  131 
  132     firstToFinish = onDistribution2 "firstToFinish" $ \x y -> x + y - x * y
  133 
  134     lastToFinish = onDistribution2 "lastToFinish" (*)

  136 instance DeltaQ DQ where
  137     type Probability DQ = Rational
  138 
  139     choice p = onDistribution2 "choice" $ \x y ->
  140         scale p x + scale (1 - p) y
  141       where
  142         scale = Piecewise.mapPieces . Poly.scale
  143 
  144     uniform a = DQ . Measure.uniform a
  145 
  146     successWithin (DQ m) = Function.eval (Measure.distribution m)
  147 
  148     failure (DQ m) = 1 - Measure.total m
  149 
  150     quantile (DQ m) p =
  151         eventuallyFromMaybe
  152         $ quantileFromMonotone (Measure.distribution m) p
  153 
  154     earliest (DQ m) = eventuallyFromMaybe $ fmap fst $ Measure.support m
  155 
  156     deadline (DQ m)= eventuallyFromMaybe $ fmap snd $ Measure.support m

  158 -- | Partial order of cumulative distribution functions.
  159 --
  160 -- @'leq' x y@ holds if and only if for all completion times @t@,
  161 -- the probability to succeed within the time @t@
  162 -- is always larger (or equal) for @x@ compared to @y@.
  163 -- In other words, @x@ has a higher probability of completing faster.
  164 --
  165 -- > x `leq` y  <=>  ∀ t. successWithin x t >= successWithin y t
  166 instance PartialOrd DQ where
  167     m1 `leq` m2 =
  168         all isNonNegativeOnSegment
  169         $ toSegments
  170         $ distribution m1 - distribution m2

  176 -- | Helper type for segements of a piecewise functions.
  177 data Segment a b
  178     = Jump a (b,b)
  179     | Polynomial (a,a) (b,b) (Poly a)
  180     | End a b
  181     deriving (Eq, Show)
  182 
  183 -- | Helper function that elaborates a piecewise function
  184 -- into a list of segments.
  185 toSegments :: (a ~ Rational) => Piecewise (Poly a) -> [Segment a a]
  186 toSegments = goJump 0 . Piecewise.toAscPieces
  187   where
  188     goJump _ [] = []
  189     goJump prev ((x1, o) : xos)
  190         | y1 - y0 > 0 = Jump x1 (y0, y1) : nexts
  191         | otherwise = nexts
  192       where
  193         y1 = Poly.eval o x1
  194         y0 = Poly.eval prev x1
  195         nexts = goPoly x1 y1 o xos
  196 
  197     goPoly x1 y1 o [] =
  198         End x1 y1 : goJump o []
  199     goPoly x1 y1 o xos@((x2, _) : _) =
  200         Polynomial (x1, x2) (y1, Poly.eval o x2) o : goJump o xos
  201         -- TODO: What about the case where y1 == y2, i.e. a constant Polynomial?
  202 
  203 {-----------------------------------------------------------------------------
  204     Operations
  205     quantile
  206 ------------------------------------------------------------------------------}
  207 -- | Compute a quantile from a monotonically increasing function.
  208 quantileFromMonotone :: (a ~ Rational) => Piecewise (Poly a) -> a -> Maybe a
  209 quantileFromMonotone pieces = findInSegments segments
  210   where
  211     segments = toSegments pieces
  212 
  213     findInSegments _ 0
  214         = Just 0
  215     findInSegments [] _
  216         = Nothing
  217     findInSegments (Jump x1 (y1, y2) : xys) y
  218         | y1 < y && y <= y2 = Just x1
  219         | otherwise = findInSegments xys y
  220     findInSegments (Polynomial (x1, x2) (y1, y2) o : xys) y
  221         | y1 < y && y <= y2 = Poly.root precision y (x1, x2) o
  222         | otherwise = findInSegments xys y
  223     findInSegments (End x1 y1 : _) y
  224         | y1 == y = Just x1
  225         | otherwise = Nothing
  226 
  227 precision :: Rational
  228 precision = 1 / 10^(10 :: Integer)

  232     meetsQTA
  233 ------------------------------------------------------------------------------}
  234 -- | Test whether the given probability distribution of completion times
  235 -- is equal to or better than a given
  236 -- __quantitative timeliness agreement__ (QTA).
  237 --
  238 -- Synonym for `leq` of the partial order,
  239 --
  240 -- > p `meetsQTA` qta  =  p `leq` qta
  241 meetsQTA :: DQ -> DQ -> Bool
  242 meetsQTA = leq
  243 
  244 isNonNegativeOnSegment :: (a ~ Rational) => Segment a a -> Bool
  245 isNonNegativeOnSegment (Jump _ (y1, y2)) =
  246     y1 >= 0 && y2 >= 0
  247 isNonNegativeOnSegment (Polynomial (x1, x2) _ poly) =
  248     compareToZero == Just GT || compareToZero == Just EQ
  249   where
  250     compareToZero = Poly.compareToZero (x1, x2, poly)
  251 isNonNegativeOnSegment (End _ y) =
  252     y >= 0

  258 -- | Compute the success probability of a 'DQ',
  259 -- and the first commonly used 'Moments' of the
  260 -- probability distribution conditioned on success.
  261 moments :: DQ -> (Rational, Moments Rational)
  262 moments (DQ m)
  263     | success == 0 =
  264         (0, Moments{mean = 0, variance = 0, skewness = 0, kurtosis = 1})
  265     | otherwise =
  266         (success, Prob.moments conditional)
  267   where
  268     success = Measure.total m
  269     conditional = Prob.unsafeFromMeasure $ Measure.scale (1/success) m

  273     Timeout
  274 ------------------------------------------------------------------------------}
  275 -- | Given a duration @dt@,
  276 -- decompose a probability distribution of completion times
  277 -- into two conditional probability distributions —
  278 -- one distribution conditional on finishing within @dt@,
  279 -- and the other distribution conditional on finishing after @dt@.
  280 --
  281 -- >   (within, p, after) = timeout dt o
  282 -- >
  283 -- > implies
  284 -- >
  285 -- >   o = choice p within (wait dt .>>. after)
  286 -- >   p = successWithin o dt
  287 -- >
  288 -- >   successWithin within dt = 1
  289 -- >     if p > 0
  290 --
  291 -- We define the corner cases as follows:
  292 --
  293 -- * @successWithin o dt = 0@ implies @before = never@.
  294 --
  295 -- * @successWithin o dt = 1@ implies @after = never@.
  296 --
  297 timeout :: Duration DQ -> DQ -> (DQ, Probability DQ, DQ)
  298 timeout dt (DQ m) = (beforeCond, p, afterCond)
  299   where
  300     before = Measure.beforeOrAt dt m
  301     after = Measure.translate (- dt) $ Measure.after dt m
  302     p = Measure.total before
  303     beforeCond
  304         | p == 0 = never
  305         | otherwise = DQ $ Measure.scale (1/p) before
  306     afterCond
  307         | p == 1 = never
  308         | otherwise = DQ $ Measure.scale (1/(1-p)) after
