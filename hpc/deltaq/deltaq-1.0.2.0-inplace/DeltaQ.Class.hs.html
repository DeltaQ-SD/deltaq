   46 -- | An 'Outcome' is the result of an activity that takes time,
   47 -- such as a distributed computation, communication, bus ride, … .
   48 --
   49 -- 'Outcome's can be composed in sequence or in parallel.
   50 class (Ord (Duration o), Num (Duration o)) => Outcome o where
   51     -- | Numerical type representing times in \( [0,+∞) \).
   52     --
   53     -- For example 'Double' or 'Rational'.
   54     type Duration o
   55 
   56     -- | The outcome that never finishes.
   57     never :: o
   58 
   59     -- | The outcome that succeeds after waiting for a fixed amount of time.
   60     wait :: Duration o -> o
   61 
   62     -- | Sequential composition:
   63     --
   64     -- First run the outcome on the left,
   65     -- then run the outcome on the right.
   66     sequentially :: o -> o -> o
   67 
   68     -- | Infix operator synonym for 'sequentially'.
   69     (.>>.) :: o -> o -> o
   70     (.>>.) = sequentially
   71 
   72     -- | Parallel composition, first to finish:
   73     --
   74     -- Run two outcomes in parallel,
   75     -- finish as soon as any one of them finishes.
   76     firstToFinish :: o -> o -> o
   77 
   78     -- | Infix operator synonym for 'firstToFinish'.
   79     (.\/.) :: o -> o -> o
   80     (.\/.) = firstToFinish
   81 
   82     -- | Parallel composiiton, last to finish:
   83     --
   84     -- Run two outcomes in parallel,
   85     -- finish after all of them have finished.
   86     lastToFinish :: o -> o -> o
   87 
   88     -- | Infix operator synonym for 'lastToFinish'.
   89     (./\.) :: o -> o -> o
   90     (./\.) = lastToFinish

   92 {-----------------------------------------------------------------------------
   93     Eventually
   94 ------------------------------------------------------------------------------}
   95 -- | 'Eventually' represents a value that either eventually occurs
   96 -- or is eventually abandoned.
   97 --
   98 -- Similar to the 'Maybe' type, but with a different 'Ord' instance:
   99 -- @Occurs x < Abandoned@ for all @x@.
  100 --
  101 data Eventually a
  102     = Occurs a
  103     | Abandoned
  104     deriving (Eq, Show)
  105 
  106 -- | For all @x@, we have @Occurs x < Abandoned@.
  107 instance Ord a => Ord (Eventually a) where
  108     compare Abandoned Abandoned = EQ
  109     compare Abandoned (Occurs _) = GT
  110     compare (Occurs _) Abandoned = LT
  111     compare (Occurs x) (Occurs y) = compare x y
  112 
  113 instance Functor Eventually where
  114     fmap _ Abandoned = Abandoned
  115     fmap f (Occurs x) = Occurs (f x)
  116 
  117 -- |
  118 -- > Abandoned <*> _ = Abandoned
  119 -- > _ <*> Abandoned = Abandoned
  120 instance Applicative Eventually where
  121     pure = Occurs
  122 
  123     Abandoned <*> Abandoned = Abandoned
  124     Abandoned <*> (Occurs _) = Abandoned
  125     (Occurs _) <*> Abandoned = Abandoned
  126     (Occurs f) <*> (Occurs y) = Occurs (f y)
  127 
  128 -- | Helper function to eliminate 'Eventually'.
  129 --
  130 -- See also: 'maybe'.
  131 eventually :: b -> (a -> b) -> Eventually a -> b
  132 eventually b _ Abandoned = b
  133 eventually _ f (Occurs x) = f x
  134 
  135 -- | Helper function that converts 'Maybe' to 'Eventually'.
  136 eventuallyFromMaybe :: Maybe a -> Eventually a
  137 eventuallyFromMaybe Nothing = Abandoned
  138 eventuallyFromMaybe (Just x) = Occurs x
  139 
  140 -- | Helper function that converts 'Eventually' to 'Maybe'.
  141 maybeFromEventually :: Eventually a -> Maybe a
  142 maybeFromEventually Abandoned = Nothing
  143 maybeFromEventually (Occurs x) = Just x

  149 -- | 'DeltaQ' — quality attenuation.
  150 --
  151 -- 'DeltaQ' is a probability distribution of time.
  152 --
  153 -- Specifically, 'DeltaQ' is the probability distribution
  154 -- of finish times for an outcome.
  155 class   ( Ord (Probability o)
  156         , Enum (Probability o)
  157         , Num (Probability o)
  158         , Fractional (Probability o)
  159         , Outcome o
  160         )
  161     => DeltaQ o
  162   where
  163     -- | Numerical type representing probabilities in \( [0,1] \).
  164     --
  165     -- For example 'Double' or 'Rational'.
  166     type Probability o
  167 
  168     -- | Left-biased random choice.
  169     --
  170     -- @choice p@ chooses the left argument with probablity @p@
  171     -- and the right argument with probability @(1-p)@.
  172     choice :: Probability o -> o -> o -> o
  173 
  174     -- | Random choice between multiple alternatives
  175     --
  176     -- @choices [(w_1, o_1), (w_2, o_2), …]@ chooses randomly between multiple
  177     -- outcomes. The probability @p_i@ for choosing the outcome @o_i@ is
  178     -- determined by the weights as @p_i = w_i / (w_1 + w_2 + …)@.
  179     --
  180     -- Note: The weights @w_i@ have the numeric type @Probability o@,
  181     -- but for convenience, and unlike probabilities,
  182     -- they are not restricted to the unit interval \( [0,1] \).
  183     choices :: [(Probability o, o)] -> o
  184     choices [] = never
  185     choices wos =
  186         foldr (uncurry choice) never
  187         $ zipWith (\wtot (w, o) -> (w / wtot, o)) ws wos
  188       where
  189         ws = scanr1 (+) (map fst wos)
  190 
  191     -- | Uniform probability distribution on a time interval.
  192     uniform :: Duration o -> Duration o -> o
  193 
  194     -- | Probability of /not/ finishing.
  195     failure :: o -> Probability o
  196 
  197     -- | Probability of finishing within the given time @t@.
  198     --
  199     -- \"Within\" is inclusive,
  200     -- i.e. this returns the probability that the finishing time is @<= t@.
  201     successWithin :: o -> Duration o -> Probability o
  202 
  203     -- | Given a probability @p@, return the smallest time @t@
  204     -- such that the probability of completing within that time
  205     -- is at least @p@.
  206     --
  207     -- Return 'Abandoned' if the given probability
  208     -- exceeds the probability of finishing.
  209     quantile :: o -> Probability o -> Eventually (Duration o)
  210 
  211     -- | The earliest finish time with non-zero probability.
  212     --
  213     -- Return 'Abandoned' if the outcome is 'never'.
  214     earliest :: o -> Eventually (Duration o)
  215 
  216     -- | The last finish time which still has non-zero probability to occur.
  217     --
  218     -- Return 'Abandoned' if arbitrarily late times are possible.
  219     deadline :: o -> Eventually (Duration o)
  220

  235 {-$properties-outcome
  236 
  237 'never'
  238 
  239 > never .>>. y = never
  240 > never ./\. y = never
  241 > never .\/. y = y
  242 >
  243 > x .>>. never = never
  244 > x ./\. never = never
  245 > x .\/. never = x
  246 
  247 'wait'
  248 
  249 > wait t .>>. wait s  =  wait (t+s)
  250 > wait t ./\. wait s  =  wait (max t s)
  251 > wait t .\/. wait s  =  wait (min t s)
  252 
  253 '(.>>.)'
  254 
  255 > (x .>>. y) .>>. z  =  x .>>. (y .>>. z)
  256 
  257 '(./\.)'
  258 
  259 > (x ./\. y) ./\. z  =  x ./\. (y ./\. z)
  260 >
  261 > x ./\. y  =  y ./\. x
  262 
  263 '(.\/.)'
  264 
  265 > (x .\/. y) .\/. z  =  x .\/. (y .\/. z)
  266 >
  267 > x .\/. y  =  y .\/. x

  271 {-$properties-deltaq
  272 
  273 'choice'
  274 
  275 > choice 1 x y = x
  276 > choice 0 x y = y
  277 >
  278 > choice p x y .>>. z  =  choice p (x .>>. z) (y .>>. z)
  279 > choice p x y ./\. z  =  choice p (x ./\. z) (y ./\. z)
  280 > choice p x y .\/. z  =  choice p (x .\/. z) (y .\/. z)
  281 
  282 'choices'
  283 
  284 > choices [] = never
  285 > choices ((w,o) : wos) = choice p o (choices wos)
  286 >   where  p = w / (w + sum (map fst wos))
  287 
  288 'uniform'
  289 
  290 > wait t .>>. uniform r s  =  uniform (t+r) (t+s)
  291 > uniform r s .>>. wait t  =  uniform (r+t) (s+t)
  292 
  293 'failure'
  294 
  295 > failure never      = 1
  296 > failure (wait t)   = 0
  297 > failure (x .>>. y) = 1 - (1 - failure x) * (1 - failure y)
  298 > failure (x ./\. y) = 1 - (1 - failure x) * (1 - failure y)
  299 > failure (x .\/. y) = failure x * failure y
  300 >
  301 > failure (choice p x y) = p * failure x + (1-p) * failure y
  302 > failure (uniform r s)  = 0
  303 
  304 'successWithin'
  305 
  306 > successWithin never    t = 0
  307 > successWithin (wait s) t = if t < s then 0 else 1
  308 >
  309 > successWithin (x ./\. y) t =
  310 >   successWithin t x * successWithin t y
  311 > successWithin (x .\/. y) t =
  312 >   1 - (1 - successWithin t x) * (1 - successWithin t y)
  313 >
  314 > successWithin (choice p x y) t =
  315 >   p * successWithin t x + (1-p) * successWithin t y
  316 > successWithin (uniform r s) t
  317 >   | t < r           = 0
  318 >   | r <= t && t < s = (t-r) / (s-r)
  319 >   | s <= t          = 1
  320 
  321 'quantile'
  322 
  323 > p <= q  implies  quantile o p <= quantile o q
  324 >
  325 > quantile x        0 = Occurs 0
  326 > quantile never    p = Abandoned       if p > 0
  327 > quantile (wait t) p = Occurs t        if p > 0
  328 >
  329 > quantile (uniform r s) p  =  r + p*(s-t)  if p > 0, r <= s
  330 
  331 'earliest'
  332 
  333 > earliest never      = Abandoned
  334 > earliest (wait t)   = Occurs t
  335 > earliest (x .>>. y) = (+) <$> earliest x <*> earliest y
  336 > earliest (x ./\. y) = max (earliest x) (earliest y)
  337 > earliest (x .\/. y) = min (earliest x) (earliest y)
  338 >
  339 > earliest (choice p x y) = min (earliest x) (earliest y)  if p ≠ 0, p ≠ 1
  340 > earliest (uniform r s)  = Occurs r   if r <= s
  341 
  342 'deadline'
  343 
  344 > deadline never      =  Abandoned
  345 > deadline (wait t)   =  Occurs t
  346 > deadline (x .>>. y) =  (+) <$> deadline x <*> deadline y
  347 > deadline (x ./\. y) =  max (deadline x) (deadline y)
  348 >
  349 > deadline (x .\/. y) =  min (deadline x) (deadline y)
  350 >   if failure x = 0 and failure y = 0
  351 >
  352 > deadline (x .\/. y) =  deadline x
  353 >   if failure x = 0 and failure y > 0
  354 >
  355 > deadline (x .\/. y) =  max (deadline x) (deadline y)
  356 >   if 1 > failure x > 0 and 1 > failure y > 0
  357 >
  358 > deadline (choice p x y) = max (deadline x) (deadline y)
  359 >   if p ≠ 0 and p ≠ 1 and failure x < 1 and failure y < 1
  360 >
  361 >
  362 > deadline (uniform r s)  = Occurs s   if r <= s
  363 
  364 -}
