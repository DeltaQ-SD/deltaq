   41 -- | The \"slack or hazard\" represents the distance between
   42 -- a reference point in (time, probability) space
   43 -- and a given 'DeltaQ'.
   44 --
   45 -- * 'Slack' represents the case where the 'DeltaQ' __meets__
   46 --   the performance requirements set by the reference point.
   47 -- * 'Hazard' represents the case where the 'DeltaQ' __fails__ to meet
   48 --   the performance requirements set by the reference point.
   49 --
   50 -- Both cases include information of how far the reference point is
   51 -- away.
   52 data SlackOrHazard o
   53     = Slack (Duration o) (Probability o)
   54     -- ^ We have some slack.
   55     -- Specifically, we have 'Duration' at the same probability as the reference,
   56     -- and 'Probability' at the same duration as the reference.
   57     | Hazard (Eventually (Duration o)) (Probability o)
   58     -- ^ We fail to meet the reference point.
   59     -- Specifically,
   60     -- we overshoot by 'Duration' at the same probability as the reference,
   61     -- and by 'Probability' at the same duration as the reference.
   62 
   63 deriving instance (Eq (Duration o), Eq (Probability o))
   64     => Eq (SlackOrHazard o)
   65 
   66 deriving instance (Show (Duration o), Show (Probability o))
   67     => Show (SlackOrHazard o)
   68 
   69 -- | Test whether the given 'SlackOrHazard' is 'Slack'.
   70 isSlack :: SlackOrHazard o -> Bool
   71 isSlack (Slack _ _) = True
   72 isSlack _ = False
   73 
   74 -- | Test whether the given 'SlackOrHazard' is 'Hazard'.
   75 isHazard :: SlackOrHazard o -> Bool
   76 isHazard (Hazard _ _) = True
   77 isHazard _ = False
   78 
   79 -- | Compute \"slack or hazard\" with respect to a given reference point.
   80 meetsRequirement
   81     :: DeltaQ o => o -> (Duration o, Probability o) -> SlackOrHazard o
   82 meetsRequirement o (t,p)
   83     | dp >= 0 = Slack dt dp
   84     | Abandoned <- t' = Hazard Abandoned (negate dp)
   85     | otherwise = Hazard (Occurs $ negate dt) (negate dp)
   86   where
   87     dp = p' - p
   88     dt = t - eventually err id t'
   89 
   90     t' = quantile o p
   91     p' = o `successWithin` t
   92 
   93     err = error "distanceToReference: inconsistency"
   94

  100 -- | Retry an outcome @n@ times, take the first instance that succeeds.
  101 --
  102 -- More specifically, @retryOverlapN n dt o@ models the completion times
  103 -- of a race of @n@ independent copies of the outcome @o@ against each other,
  104 -- succeeding when any of the copies first succeeds.
  105 --
  106 -- Copy number @0 <= j < n@ is started after waiting a time @j * dt@.
  107 --
  108 -- If no copy succeeds within total time @n * dt@,
  109 -- the whole outcome fails.
  110 retryOverlapN :: Int -> Duration DQ -> DQ -> DQ
  111 retryOverlapN n dt = retryOverlap (replicate n dt)
  112 
  113 -- | Retry an outcome multiple times, take the first instance that succeeds.
  114 --
  115 -- More specifically, @retryOverlap dts o@ models the completion times
  116 -- of a race of @length dts@ independent copies of the outcome @o@
  117 -- against each other, succeeding when any of the copies first succeeds.
  118 --
  119 -- Each copy is started after a timeout
  120 -- in the event that the previous copies have not finished yet;
  121 -- the list @dts@ gives the successive timeouts.
  122 --
  123 -- If no copy succeeds within total time @sum dts@,
  124 -- the whole outcome fails.
  125 retryOverlap :: [Duration DQ] -> DQ -> DQ
  126 retryOverlap dts0 o = go dts0 o
  127   where
  128     go [] _ = never
  129     go (dt:dts) race =
  130         choice p before (wait dt .>>. retry)
  131       where
  132         (before, p, after) = timeout dt race
  133 
  134         -- add another copy of `o` to the race
  135         retry = go dts (after .\/. o)
